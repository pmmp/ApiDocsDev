214 protected function storeLocal(string $key, $complexData) : void{
215 if(self::$threadLocalStorage === null){
216 /*
217 * It's necessary to use an object (not array) here because pthreads is stupid. Non-default array statics
218 * will be inherited when task classes are copied to the worker thread, which would cause unwanted
219 * inheritance of primitive thread-locals, which we really don't want for various reasons.
220 * It won't try to inherit objects though, so this is the easiest solution.
221 */
222 self::$threadLocalStorage = new \ArrayObject();
223 }
224 self::$threadLocalStorage[spl_object_id($this)][$key] = $complexData;
225 }
