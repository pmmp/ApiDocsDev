106 private function recalculateSignalStrength() : int{
107 $lightLevel = $this->position->getWorld()->getRealBlockSkyLightAt($this->position->x, $this->position->y, $this->position->z);
108 if($this->inverted){
109 return 15 - $lightLevel;
110 }
111
112 $sunAngle = $this->position->getWorld()->getSunAnglePercentage();
113 return max(0, (int) round($lightLevel * cos(($sunAngle + ((($sunAngle < 0.5 ? 0 : 1) - $sunAngle) / 5)) * 2 * M_PI)));
114 }
