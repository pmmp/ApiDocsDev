236 public static function getStoneSlabIdentifier(int $stoneSlabId, int $meta) : BlockIdentifierFlattened{
237 $id = [
238 1 => [Ids::STONE_SLAB, Ids::DOUBLE_STONE_SLAB],
239 2 => [Ids::STONE_SLAB2, Ids::DOUBLE_STONE_SLAB2],
240 3 => [Ids::STONE_SLAB3, Ids::DOUBLE_STONE_SLAB3],
241 4 => [Ids::STONE_SLAB4, Ids::DOUBLE_STONE_SLAB4]
242 ][$stoneSlabId] ?? null;
243 if($id === null){
244 throw new \InvalidArgumentException("Stone slab type should be 1, 2, 3 or 4");
245 }
246 return new BlockIdentifierFlattened($id[0], [$id[1]], $meta);
247 }
