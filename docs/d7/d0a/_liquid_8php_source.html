173 public function getFlowVector() : Vector3{
174 if($this->flowVector !== null){
175 return $this->flowVector;
176 }
177
178 $vX = $vY = $vZ = 0;
179
180 $decay = $this->getEffectiveFlowDecay($this);
181
182 $world = $this->position->getWorld();
183
184 foreach(Facing::HORIZONTAL as $j){
185 $x = $this->position->x;
186 $y = $this->position->y;
187 $z = $this->position->z;
188
189 match($j){
190 Facing::WEST => --$x,
191 Facing::EAST => ++$x,
192 Facing::NORTH => --$z,
193 Facing::SOUTH => ++$z
194 };
195
196 $sideBlock = $world->getBlockAt($x, $y, $z);
197 $blockDecay = $this->getEffectiveFlowDecay($sideBlock);
198
199 if($blockDecay < 0){
200 if(!$sideBlock->canBeFlowedInto()){
201 continue;
202 }
203
204 $blockDecay = $this->getEffectiveFlowDecay($world->getBlockAt($x, $y - 1, $z));
205
206 if($blockDecay >= 0){
207 $realDecay = $blockDecay - ($decay - 8);
208 $vX += ($x - $this->position->x) * $realDecay;
209 $vY += ($y - $this->position->y) * $realDecay;
210 $vZ += ($z - $this->position->z) * $realDecay;
211 }
212
213 continue;
214 }else{
215 $realDecay = $blockDecay - $decay;
216 $vX += ($x - $this->position->x) * $realDecay;
217 $vY += ($y - $this->position->y) * $realDecay;
218 $vZ += ($z - $this->position->z) * $realDecay;
219 }
220 }
221
222 $vector = new Vector3($vX, $vY, $vZ);
223
224 if($this->falling){
225 foreach(Facing::HORIZONTAL as $facing){
226 $pos = $this->position->getSide($facing);
227 if(
228 !$this->canFlowInto($world->getBlockAt($pos->x, $pos->y, $pos->z)) ||
229 !$this->canFlowInto($world->getBlockAt($pos->x, $pos->y + 1, $pos->z))
230 ){
231 $vector = $vector->normalize()->add(0, -6, 0);
232 break;
233 }
234 }
235 }
236
237 return $this->flowVector = $vector->normalize();
238 }

270 public function onScheduledUpdate() : void{
271 $multiplier = $this->getFlowDecayPerBlock();
272
273 $world = $this->position->getWorld();
274
275 if(!$this->isSource()){
276 $smallestFlowDecay = -100;
277 $this->adjacentSources = 0;
278 $smallestFlowDecay = $this->getSmallestFlowDecay($world->getBlockAt($this->position->x, $this->position->y, $this->position->z - 1), $smallestFlowDecay);
279 $smallestFlowDecay = $this->getSmallestFlowDecay($world->getBlockAt($this->position->x, $this->position->y, $this->position->z + 1), $smallestFlowDecay);
280 $smallestFlowDecay = $this->getSmallestFlowDecay($world->getBlockAt($this->position->x - 1, $this->position->y, $this->position->z), $smallestFlowDecay);
281 $smallestFlowDecay = $this->getSmallestFlowDecay($world->getBlockAt($this->position->x + 1, $this->position->y, $this->position->z), $smallestFlowDecay);
282
283 $newDecay = $smallestFlowDecay + $multiplier;
284 $falling = false;
285
286 if($newDecay > self::MAX_DECAY || $smallestFlowDecay < 0){
287 $newDecay = -1;
288 }
289
290 if($this->getEffectiveFlowDecay($world->getBlockAt($this->position->x, $this->position->y + 1, $this->position->z)) >= 0){
291 $falling = true;
292 }
293
294 $minAdjacentSources = $this->getMinAdjacentSourcesToFormSource();
295 if($minAdjacentSources !== null && $this->adjacentSources >= $minAdjacentSources){
296 $bottomBlock = $world->getBlockAt($this->position->x, $this->position->y - 1, $this->position->z);
297 if($bottomBlock->isSolid() || ($bottomBlock instanceof Liquid && $bottomBlock->isSameType($this) && $bottomBlock->isSource())){
298 $newDecay = 0;
299 $falling = false;
300 }
301 }
302
303 if($falling !== $this->falling || (!$falling && $newDecay !== $this->decay)){
304 if(!$falling && $newDecay < 0){
305 $world->setBlock($this->position, VanillaBlocks::AIR());
306 return;
307 }
308
309 $this->falling = $falling;
310 $this->decay = $falling ? 0 : $newDecay;
311 $world->setBlock($this->position, $this); //local block update will cause an update to be scheduled
312 }
313 }
314
315 $bottomBlock = $world->getBlockAt($this->position->x, $this->position->y - 1, $this->position->z);
316
317 $this->flowIntoBlock($bottomBlock, 0, true);
318
319 if($this->isSource() || !$bottomBlock->canBeFlowedInto()){
320 if($this->falling){
321 $adjacentDecay = 1; //falling liquid behaves like source block
322 }else{
323 $adjacentDecay = $this->decay + $multiplier;
324 }
325
326 if($adjacentDecay <= self::MAX_DECAY){
327 $calculator = new MinimumCostFlowCalculator($this->position->getWorld(), $this->getFlowDecayPerBlock(), \Closure::fromCallable([$this, 'canFlowInto']));
328 foreach($calculator->getOptimalFlowDirections($this->position->getFloorX(), $this->position->getFloorY(), $this->position->getFloorZ()) as $facing){
329 $this->flowIntoBlock($world->getBlock($this->position->getSide($facing)), $adjacentDecay, false);
330 }
331 }
332 }
333
334 $this->checkForHarden();
335 }

337 protected function flowIntoBlock(Block $block, int $newFlowDecay, bool $falling) : void{
338 if($this->canFlowInto($block) && !($block instanceof Liquid)){
339 $new = clone $this;
340 $new->falling = $falling;
341 $new->decay = $falling ? 0 : $newFlowDecay;
342
343 $ev = new BlockSpreadEvent($block, $this, $new);
344 $ev->call();
345 if(!$ev->isCancelled()){
346 if($block->getId() > 0){
347 $this->position->getWorld()->useBreakOn($block->position);
348 }
349
350 $this->position->getWorld()->setBlock($block->position, $ev->getNewState());
351 }
352 }

356 private function getSmallestFlowDecay(Block $block, int $decay) : int{
357 if(!($block instanceof Liquid) || !$block->isSameType($this)){
358 return $decay;
359 }
360
361 $blockDecay = $block->decay;
362
363 if($block->isSource()){
364 ++$this->adjacentSources;
365 }elseif($block->falling){
366 $blockDecay = 0;
367 }
368
369 return ($decay >= 0 && $blockDecay >= $decay) ? $decay : $blockDecay;
370 }
