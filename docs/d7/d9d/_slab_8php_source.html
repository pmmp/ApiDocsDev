41 public function __construct(BlockIdentifierFlattened $idInfo, string $name, BlockBreakInfo $breakInfo){
42 $this->idInfoFlattened = $idInfo;
43 parent::__construct($idInfo, $name . " Slab", $breakInfo);
44 $this->slabType = SlabType::BOTTOM();
45 }
46
47 public function getId() : int{
48 return $this->slabType->equals(SlabType::DOUBLE()) ? $this->idInfoFlattened->getSecondId() : parent::getId();
49 }
50
51 protected function writeStateToMeta() : int{
52 if(!$this->slabType->equals(SlabType::DOUBLE())){
53 return ($this->slabType->equals(SlabType::TOP()) ? BlockLegacyMetadata::SLAB_FLAG_UPPER : 0);
54 }
55 return 0;
56 }
57
58 public function readStateFromData(int $id, int $stateMeta) : void{
59 if($id === $this->idInfoFlattened->getSecondId()){
60 $this->slabType = SlabType::DOUBLE();
61 }else{
62 $this->slabType = ($stateMeta & BlockLegacyMetadata::SLAB_FLAG_UPPER) !== 0 ? SlabType::TOP() : SlabType::BOTTOM();
63 }
64 }
65
66 public function getStateBitmask() : int{
67 return 0b1000;
68 }
69
70 public function isTransparent() : bool{
71 return !$this->slabType->equals(SlabType::DOUBLE());
72 }

89 public function canBePlacedAt(Block $blockReplace, Vector3 $clickVector, int $face, bool $isClickedBlock) : bool{
90 if(parent::canBePlacedAt($blockReplace, $clickVector, $face, $isClickedBlock)){
91 return true;
92 }
93
94 if($blockReplace instanceof Slab && !$blockReplace->slabType->equals(SlabType::DOUBLE()) && $blockReplace->isSameType($this)){
95 if($blockReplace->slabType->equals(SlabType::TOP())){ //Trying to combine with top slab
96 return $clickVector->y <= 0.5 || (!$isClickedBlock && $face === Facing::UP);
97 }else{
98 return $clickVector->y >= 0.5 || (!$isClickedBlock && $face === Facing::DOWN);
99 }
100 }
101
102 return false;
103 }

105 public function place(BlockTransaction $tx, Item $item, Block $blockReplace, Block $blockClicked, int $face, Vector3 $clickVector, ?Player $player = null) : bool{
106 if($blockReplace instanceof Slab && !$blockReplace->slabType->equals(SlabType::DOUBLE()) && $blockReplace->isSameType($this) && (
107 ($blockReplace->slabType->equals(SlabType::TOP()) && ($clickVector->y <= 0.5 || $face === Facing::UP)) ||
108 ($blockReplace->slabType->equals(SlabType::BOTTOM()) && ($clickVector->y >= 0.5 || $face === Facing::DOWN))
109 )){
110 //Clicked in empty half of existing slab
111 $this->slabType = SlabType::DOUBLE();
112 }else{
113 $this->slabType = (($face !== Facing::UP && $clickVector->y > 0.5) || $face === Facing::DOWN) ? SlabType::TOP() : SlabType::BOTTOM();
114 }
115
116 return parent::place($tx, $item, $blockReplace, $blockClicked, $face, $clickVector, $player);
117 }

122 protected function recalculateCollisionBoxes() : array{
123 if($this->slabType->equals(SlabType::DOUBLE())){
124 return [AxisAlignedBB::one()];
125 }
126 return [AxisAlignedBB::one()->trim($this->slabType->equals(SlabType::TOP()) ? Facing::DOWN : Facing::UP, 0.5)];
127 }
128
129 public function getSupportType(int $facing) : SupportType{
130 if($this->slabType->equals(SlabType::DOUBLE())){
131 return SupportType::FULL();
132 }elseif(($facing === Facing::UP && $this->slabType->equals(SlabType::TOP())) || ($facing === Facing::DOWN && $this->slabType->equals(SlabType::BOTTOM()))){
133 return SupportType::FULL();
134 }
135 return SupportType::NONE();
136 }
137
138 public function getDropsForCompatibleTool(Item $item) : array{
139 return [$this->asItem()->setCount($this->slabType->equals(SlabType::DOUBLE()) ? 2 : 1)];
140 }
