133 public function updateText(Player $author, SignText $text) : bool{
134 $size = 0;
135 foreach($text->getLines() as $line){
136 $size += strlen($line);
137 }
138 if($size > 1000){
139 throw new \UnexpectedValueException($author->getName() . " tried to write $size bytes of text onto a sign (bigger than max 1000)");
140 }
141 $ev = new SignChangeEvent($this, $author, new SignText(array_map(function(string $line) : string{
142 return TextFormat::clean($line, false);
143 }, $text->getLines())));
144 if($this->editorEntityRuntimeId === null || $this->editorEntityRuntimeId !== $author->getId()){
145 $ev->cancel();
146 }
147 $ev->call();
148 if(!$ev->isCancelled()){
149 $this->setText($ev->getNewText());
150 $this->position->getWorld()->setBlock($this->position, $this);
151 return true;
152 }
153
154 return false;
155 }
