141 public function onInteract(Item $item, int $face, Vector3 $clickVector, ?Player $player = null, array &$returnedItems = []) : bool{
142 if($player === null){
143 return false;
144 }
145 $dyeColor = $item instanceof Dye ? $item->getColor() : match($item->getTypeId()){
146 ItemTypeIds::BONE_MEAL => DyeColor::WHITE(),
147 ItemTypeIds::LAPIS_LAZULI => DyeColor::BLUE(),
148 ItemTypeIds::COCOA_BEANS => DyeColor::BROWN(),
149 default => null
150 };
151 if($dyeColor !== null){
152 $color = $dyeColor->equals(DyeColor::BLACK()) ? new Color(0, 0, 0) : $dyeColor->getRgbValue();
153 if($color->toARGB() === $this->text->getBaseColor()->toARGB()){
154 return false;
155 }
156
157 if($this->doSignChange(new SignText($this->text->getLines(), $color, $this->text->isGlowing()), $player, $item)){
158 $this->position->getWorld()->addSound($this->position, new DyeUseSound());
159 return true;
160 }
161 }elseif($item->getTypeId() === ItemTypeIds::INK_SAC){
162 return $this->changeSignGlowingState(false, $player, $item);
163 }elseif($item->getTypeId() === ItemTypeIds::GLOW_INK_SAC){
164 return $this->changeSignGlowingState(true, $player, $item);
165 }
166 return false;
167 }

188 public function updateText(Player $author, SignText $text) : bool{
189 $size = 0;
190 foreach($text->getLines() as $line){
191 $size += strlen($line);
192 }
193 if($size > 1000){
194 throw new \UnexpectedValueException($author->getName() . " tried to write $size bytes of text onto a sign (bigger than max 1000)");
195 }
196 $ev = new SignChangeEvent($this, $author, new SignText(array_map(function(string $line) : string{
197 return TextFormat::clean($line, false);
198 }, $text->getLines())));
199 if($this->editorEntityRuntimeId === null || $this->editorEntityRuntimeId !== $author->getId()){
200 $ev->cancel();
201 }
202 $ev->call();
203 if(!$ev->isCancelled()){
204 $this->setText($ev->getNewText());
205 $this->position->getWorld()->setBlock($this->position, $this);
206 return true;
207 }
208
209 return false;
210 }
