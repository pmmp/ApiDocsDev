114 public function updateText(Player $author, SignText $text) : bool{
115 $size = 0;
116 foreach($text->getLines() as $line){
117 $size += strlen($line);
118 }
119 if($size > 1000){
120 throw new \UnexpectedValueException($author->getName() . " tried to write $size bytes of text onto a sign (bigger than max 1000)");
121 }
122 $ev = new SignChangeEvent($this, $author, new SignText(array_map(function(string $line) : string{
123 return TextFormat::clean($line, false);
124 }, $text->getLines())));
125 if($this->editorEntityRuntimeId === null || $this->editorEntityRuntimeId !== $author->getId()){
126 $ev->cancel();
127 }
128 $ev->call();
129 if(!$ev->isCancelled()){
130 $this->setText($ev->getNewText());
131 $this->position->getWorld()->setBlock($this->position, $this);
132 return true;
133 }
134
135 return false;
136 }
