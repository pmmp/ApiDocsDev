46 protected function writeStateToMeta() : int{
47 $rotationMeta = match($this->facing->id()){
48 LeverFacing::DOWN_AXIS_X()->id() => 0,
49 LeverFacing::EAST()->id() => 1,
50 LeverFacing::WEST()->id() => 2,
51 LeverFacing::SOUTH()->id() => 3,
52 LeverFacing::NORTH()->id() => 4,
53 LeverFacing::UP_AXIS_Z()->id() => 5,
54 LeverFacing::UP_AXIS_X()->id() => 6,
55 LeverFacing::DOWN_AXIS_Z()->id() => 7,
56 default => throw new AssumptionFailedError(),
57 };
58 return $rotationMeta | ($this->activated ? BlockLegacyMetadata::LEVER_FLAG_POWERED : 0);
59 }

61 public function readStateFromData(int $id, int $stateMeta) : void{
62 $rotationMeta = $stateMeta & 0x07;
63 $this->facing = match($rotationMeta){
64 0 => LeverFacing::DOWN_AXIS_X(),
65 1 => LeverFacing::EAST(),
66 2 => LeverFacing::WEST(),
67 3 => LeverFacing::SOUTH(),
68 4 => LeverFacing::NORTH(),
69 5 => LeverFacing::UP_AXIS_Z(),
70 6 => LeverFacing::UP_AXIS_X(),
71 7 => LeverFacing::DOWN_AXIS_Z(),
72 default => throw new AssumptionFailedError("0x07 mask should make this impossible"), //phpstan doesn't understand :(
73 };
74
75 $this->activated = ($stateMeta & BlockLegacyMetadata::LEVER_FLAG_POWERED) !== 0;
76 }

98 public function place(BlockTransaction $tx, Item $item, Block $blockReplace, Block $blockClicked, int $face, Vector3 $clickVector, ?Player $player = null) : bool{
99 if(!$this->canBeSupportedBy($blockClicked, $face)){
100 return false;
101 }
102
103 $selectUpDownPos = function(LeverFacing $x, LeverFacing $z) use ($player) : LeverFacing{
104 if($player !== null){
105 return Facing::axis($player->getHorizontalFacing()) === Axis::X ? $x : $z;
106 }
107 return $x;
108 };
109 $this->facing = match($face){
110 Facing::DOWN => $selectUpDownPos(LeverFacing::DOWN_AXIS_X(), LeverFacing::DOWN_AXIS_Z()),
111 Facing::UP => $selectUpDownPos(LeverFacing::UP_AXIS_X(), LeverFacing::UP_AXIS_Z()),
112 Facing::NORTH => LeverFacing::NORTH(),
113 Facing::SOUTH => LeverFacing::SOUTH(),
114 Facing::WEST => LeverFacing::WEST(),
115 Facing::EAST => LeverFacing::EAST(),
116 default => throw new AssumptionFailedError("Bad facing value"),
117 };
118
119 return parent::place($tx, $item, $blockReplace, $blockClicked, $face, $clickVector, $player);
120 }

125 $this->position->getWorld()->useBreakOn($this->position);

129 public function onInteract(Item $item, int $face, Vector3 $clickVector, ?Player $player = null) : bool{
130 $this->activated = !$this->activated;
131 $this->position->getWorld()->setBlock($this->position, $this);
132 $this->position->getWorld()->addSound(
133 $this->position->add(0.5, 0.5, 0.5),
134 $this->activated ? new RedstonePowerOnSound() : new RedstonePowerOffSound()
135 );
136 return true;
137 }
