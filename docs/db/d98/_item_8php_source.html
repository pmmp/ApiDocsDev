272 protected function deserializeCompoundTag(CompoundTag $tag) : void{
273 $this->customName = "";
274 $this->lore = [];
275
276 $display = $tag->getCompoundTag(self::TAG_DISPLAY);
277 if($display !== null){
278 $this->customName = $display->getString(self::TAG_DISPLAY_NAME, $this->customName);
279 $lore = $display->getListTag(self::TAG_DISPLAY_LORE);
280 if($lore !== null && $lore->getTagType() === NBT::TAG_String){
282 foreach($lore as $t){
283 $this->lore[] = $t->getValue();
284 }
285 }
286 }
287
288 $this->removeEnchantments();
289 $enchantments = $tag->getListTag(self::TAG_ENCH);
290 if($enchantments !== null && $enchantments->getTagType() === NBT::TAG_Compound){
292 foreach($enchantments as $enchantment){
293 $magicNumber = $enchantment->getShort("id", -1);
294 $level = $enchantment->getShort("lvl", 0);
295 if($level <= 0){
296 continue;
297 }
298 $type = EnchantmentIdMap::getInstance()->fromId($magicNumber);
299 if($type !== null){
300 $this->addEnchantment(new EnchantmentInstance($type, $level));
301 }
302 }
303 }
304
305 $this->blockEntityTag = $tag->getCompoundTag(self::TAG_BLOCK_ENTITY_TAG);
306
307 $this->canPlaceOn = [];
308 $canPlaceOn = $tag->getListTag("CanPlaceOn");
309 if($canPlaceOn !== null){
311 foreach($canPlaceOn as $entry){
312 $this->canPlaceOn[$entry->getValue()] = $entry->getValue();
313 }
314 }
315 $this->canDestroy = [];
316 $canDestroy = $tag->getListTag("CanDestroy");
317 if($canDestroy !== null){
319 foreach($canDestroy as $entry){
320 $this->canDestroy[$entry->getValue()] = $entry->getValue();
321 }
322 }
323 }

325 protected function serializeCompoundTag(CompoundTag $tag) : void{
326 $display = $tag->getCompoundTag(self::TAG_DISPLAY) ?? new CompoundTag();
327
328 $this->hasCustomName() ?
329 $display->setString(self::TAG_DISPLAY_NAME, $this->getCustomName()) :
330 $display->removeTag(self::TAG_DISPLAY_NAME);
331
332 if(count($this->lore) > 0){
333 $loreTag = new ListTag();
334 foreach($this->lore as $line){
335 $loreTag->push(new StringTag($line));
336 }
337 $display->setTag(self::TAG_DISPLAY_LORE, $loreTag);
338 }else{
339 $display->removeTag(self::TAG_DISPLAY_LORE);
340 }
341 $display->count() > 0 ?
342 $tag->setTag(self::TAG_DISPLAY, $display) :
343 $tag->removeTag(self::TAG_DISPLAY);
344
345 if($this->hasEnchantments()){
346 $ench = new ListTag();
347 foreach($this->getEnchantments() as $enchantmentInstance){
348 $ench->push(CompoundTag::create()
349 ->setShort("id", EnchantmentIdMap::getInstance()->toId($enchantmentInstance->getType()))
350 ->setShort("lvl", $enchantmentInstance->getLevel())
351 );
352 }
353 $tag->setTag(self::TAG_ENCH, $ench);
354 }else{
355 $tag->removeTag(self::TAG_ENCH);
356 }
357
358 ($blockData = $this->getCustomBlockData()) !== null ?
359 $tag->setTag(self::TAG_BLOCK_ENTITY_TAG, clone $blockData) :
360 $tag->removeTag(self::TAG_BLOCK_ENTITY_TAG);
361
362 if(count($this->canPlaceOn) > 0){
363 $canPlaceOn = new ListTag();
364 foreach($this->canPlaceOn as $item){
365 $canPlaceOn->push(new StringTag($item));
366 }
367 $tag->setTag("CanPlaceOn", $canPlaceOn);
368 }else{
369 $tag->removeTag("CanPlaceOn");
370 }
371 if(count($this->canDestroy) > 0){
372 $canDestroy = new ListTag();
373 foreach($this->canDestroy as $item){
374 $canDestroy->push(new StringTag($item));
375 }
376 $tag->setTag("CanDestroy", $canDestroy);
377 }else{
378 $tag->removeTag("CanDestroy");
379 }

636 $nbt = "";

679 public static function nbtDeserialize(CompoundTag $tag) : Item{
680 if($tag->getTag("id") === null || $tag->getTag("Count") === null){
681 return ItemFactory::getInstance()->get(0);
682 }
683
684 $count = Binary::unsignByte($tag->getByte("Count"));
685 $meta = $tag->getShort("Damage", 0);
686
687 $idTag = $tag->getTag("id");
688 if($idTag instanceof ShortTag){
689 $item = ItemFactory::getInstance()->get($idTag->getValue(), $meta, $count);
690 }elseif($idTag instanceof StringTag){ //PC item save format
691 try{
692 $item = LegacyStringToItemParser::getInstance()->parse($idTag->getValue() . ":$meta");
693 }catch(LegacyStringToItemParserException $e){
694 //TODO: improve error handling
695 return VanillaItems::AIR();
696 }
697 $item->setCount($count);
698 }else{
699 throw new SavedDataLoadingException("Item CompoundTag ID must be an instance of StringTag or ShortTag, " . get_class($idTag) . " given");
700 }
701
702 $itemNBT = $tag->getCompoundTag("tag");
703 if($itemNBT !== null){
704 $item->setNamedTag(clone $itemNBT);
705 }
706
707 return $item;
708 }
