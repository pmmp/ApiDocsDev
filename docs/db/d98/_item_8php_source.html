285 protected function deserializeCompoundTag(CompoundTag $tag) : void{
286 $this->customName = "";
287 $this->lore = [];
288
289 $display = $tag->getCompoundTag(self::TAG_DISPLAY);
290 if($display !== null){
291 $this->customName = $display->getString(self::TAG_DISPLAY_NAME, $this->customName);
292 $lore = $display->getListTag(self::TAG_DISPLAY_LORE);
293 if($lore !== null && $lore->getTagType() === NBT::TAG_String){
295 foreach($lore as $t){
296 $this->lore[] = $t->getValue();
297 }
298 }
299 }
300
301 $this->removeEnchantments();
302 $enchantments = $tag->getListTag(self::TAG_ENCH);
303 if($enchantments !== null && $enchantments->getTagType() === NBT::TAG_Compound){
305 foreach($enchantments as $enchantment){
306 $magicNumber = $enchantment->getShort(self::TAG_ENCH_ID, -1);
307 $level = $enchantment->getShort(self::TAG_ENCH_LVL, 0);
308 if($level <= 0){
309 continue;
310 }
311 $type = EnchantmentIdMap::getInstance()->fromId($magicNumber);
312 if($type !== null){
313 $this->addEnchantment(new EnchantmentInstance($type, $level));
314 }
315 }
316 }
317
318 $this->blockEntityTag = $tag->getCompoundTag(self::TAG_BLOCK_ENTITY_TAG);
319
320 $this->canPlaceOn = [];
321 $canPlaceOn = $tag->getListTag(self::TAG_CAN_PLACE_ON);
322 if($canPlaceOn !== null && $canPlaceOn->getTagType() === NBT::TAG_String){
324 foreach($canPlaceOn as $entry){
325 $this->canPlaceOn[$entry->getValue()] = $entry->getValue();
326 }
327 }
328 $this->canDestroy = [];
329 $canDestroy = $tag->getListTag(self::TAG_CAN_DESTROY);
330 if($canDestroy !== null && $canDestroy->getTagType() === NBT::TAG_String){
332 foreach($canDestroy as $entry){
333 $this->canDestroy[$entry->getValue()] = $entry->getValue();
334 }
335 }
336
337 $this->keepOnDeath = $tag->getByte(self::TAG_KEEP_ON_DEATH, 0) !== 0;
338 }

340 protected function serializeCompoundTag(CompoundTag $tag) : void{
341 $display = $tag->getCompoundTag(self::TAG_DISPLAY);
342
343 if($this->customName !== ""){
344 $display ??= new CompoundTag();
345 $display->setString(self::TAG_DISPLAY_NAME, $this->customName);
346 }else{
347 $display?->removeTag(self::TAG_DISPLAY_NAME);
348 }
349
350 if(count($this->lore) > 0){
351 $loreTag = new ListTag();
352 foreach($this->lore as $line){
353 $loreTag->push(new StringTag($line));
354 }
355 $display ??= new CompoundTag();
356 $display->setTag(self::TAG_DISPLAY_LORE, $loreTag);
357 }else{
358 $display?->removeTag(self::TAG_DISPLAY_LORE);
359 }
360 $display !== null && $display->count() > 0 ?
361 $tag->setTag(self::TAG_DISPLAY, $display) :
362 $tag->removeTag(self::TAG_DISPLAY);
363
364 if(count($this->enchantments) > 0){
365 $ench = new ListTag();
366 $enchantmentIdMap = EnchantmentIdMap::getInstance();
367 foreach($this->enchantments as $enchantmentInstance){
368 $ench->push(CompoundTag::create()
369 ->setShort(self::TAG_ENCH_ID, $enchantmentIdMap->toId($enchantmentInstance->getType()))
370 ->setShort(self::TAG_ENCH_LVL, $enchantmentInstance->getLevel())
371 );
372 }
373 $tag->setTag(self::TAG_ENCH, $ench);
374 }else{
375 $tag->removeTag(self::TAG_ENCH);
376 }
377
378 $this->blockEntityTag !== null ?
379 $tag->setTag(self::TAG_BLOCK_ENTITY_TAG, clone $this->blockEntityTag) :
380 $tag->removeTag(self::TAG_BLOCK_ENTITY_TAG);
381
382 if(count($this->canPlaceOn) > 0){
383 $canPlaceOn = new ListTag();
384 foreach($this->canPlaceOn as $item){
385 $canPlaceOn->push(new StringTag($item));
386 }
387 $tag->setTag(self::TAG_CAN_PLACE_ON, $canPlaceOn);
388 }else{
389 $tag->removeTag(self::TAG_CAN_PLACE_ON);
390 }
391 if(count($this->canDestroy) > 0){
392 $canDestroy = new ListTag();
393 foreach($this->canDestroy as $item){
394 $canDestroy->push(new StringTag($item));
395 }
396 $tag->setTag(self::TAG_CAN_DESTROY, $canDestroy);
397 }else{
398 $tag->removeTag(self::TAG_CAN_DESTROY);
399 }
400
401 if($this->keepOnDeath){
402 $tag->setByte(self::TAG_KEEP_ON_DEATH, 1);
403 }else{
404 $tag->removeTag(self::TAG_KEEP_ON_DEATH);
405 }

675 final public static function legacyJsonDeserialize(array $data) : Item{
676 $nbt = "";
677
678 //Backwards compatibility
679 if(isset($data["nbt"])){
680 $nbt = $data["nbt"];
681 }elseif(isset($data["nbt_hex"])){
682 $nbt = hex2bin($data["nbt_hex"]);
683 }elseif(isset($data["nbt_b64"])){
684 $nbt = base64_decode($data["nbt_b64"], true);
685 }
686
687 $itemStackData = GlobalItemDataHandlers::getUpgrader()->upgradeItemTypeDataInt(
688 (int) $data["id"],
689 (int) ($data["damage"] ?? 0),
690 (int) ($data["count"] ?? 1),
691 $nbt !== "" ? (new LittleEndianNbtSerializer())->read($nbt)->mustGetCompoundTag() : null
692 );
693
694 try{
695 return GlobalItemDataHandlers::getDeserializer()->deserializeStack($itemStackData);
696 }catch(ItemTypeDeserializeException $e){
697 throw new SavedDataLoadingException($e->getMessage(), 0, $e);
698 }
699 }
