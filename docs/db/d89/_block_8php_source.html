147

619 public function calculateIntercept(Vector3 $pos1, Vector3 $pos2) : ?RayTraceResult{
620 $bbs = $this->getCollisionBoxes();
621 if(count($bbs) === 0){
622 return null;
623 }
624
626 $currentHit = null;
628 $currentDistance = PHP_INT_MAX;
629
630 foreach($bbs as $bb){
631 $nextHit = $bb->calculateIntercept($pos1, $pos2);
632 if($nextHit === null){
633 continue;
634 }
635
636 $nextDistance = $nextHit->hitVector->distanceSquared($pos1);
637 if($nextDistance < $currentDistance){
638 $currentHit = $nextHit;
639 $currentDistance = $nextDistance;
640 }
641 }
642
643 return $currentHit;
644 }
