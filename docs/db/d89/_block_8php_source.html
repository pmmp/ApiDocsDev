123 final public function decodeTypeData(int $data) : void{
124 $typeBits = $this->getRequiredTypeDataBits();
125 $givenBits = $typeBits;
126 $reader = new RuntimeDataReader($givenBits, $data);
127
128 $this->describeType($reader);
129 $readBits = $reader->getOffset();
130 if($typeBits !== $readBits){
131 throw new \LogicException(get_class($this) . ": Exactly $typeBits bits of type data were provided, but $readBits were read");
132 }

138 final public function decodeStateData(int $data) : void{
139 $typeBits = $this->getRequiredTypeDataBits();
140 $stateBits = $this->getRequiredStateDataBits();
141 $givenBits = $typeBits + $stateBits;
142 $reader = new RuntimeDataReader($givenBits, $data);
143 $this->decodeTypeData($reader->readInt($typeBits));
144
145 $this->describeState($reader);
146 $readBits = $reader->getOffset() - $typeBits;
147 if($stateBits !== $readBits){
148 throw new \LogicException(get_class($this) . ": Exactly $stateBits bits of state data were provided, but $readBits were read");
149 }
150 }

155 final public function computeTypeData() : int{
156 $typeBits = $this->getRequiredTypeDataBits();
157 $requiredBits = $typeBits;
158 $writer = new RuntimeDataWriter($requiredBits);
159
160 $this->describeType($writer);
161 $writtenBits = $writer->getOffset();
162 if($typeBits !== $writtenBits){
163 throw new \LogicException(get_class($this) . ": Exactly $typeBits bits of type data were expected, but $writtenBits were written");
164 }
165
166 return $writer->getValue();
167 }

172 final public function computeStateData() : int{
173 $typeBits = $this->getRequiredTypeDataBits();
174 $stateBits = $this->getRequiredStateDataBits();
175 $requiredBits = $typeBits + $stateBits;
176 $writer = new RuntimeDataWriter($requiredBits);
177 $writer->int($typeBits, $this->computeTypeData());
178
179 $this->describeState($writer);
180 $writtenBits = $writer->getOffset() - $typeBits;
181 if($stateBits !== $writtenBits){
182 throw new \LogicException(get_class($this) . ": Exactly $stateBits bits of state data were expected, but $writtenBits were written");
183 }
184
185 return $writer->getValue();
186 }

218 public function writeStateToWorld() : void{
219 $world = $this->position->getWorld();
220 $chunk = $world->getOrLoadChunkAtPosition($this->position);
221 if($chunk === null){
222 throw new AssumptionFailedError("World::setBlock() should have loaded the chunk before calling this method");
223 }
224 $chunk->setBlockStateId($this->position->x & Chunk::COORD_MASK, $this->position->y, $this->position->z & Chunk::COORD_MASK, $this->getStateId());
225
226 $tileType = $this->idInfo->getTileClass();
227 $oldTile = $world->getTile($this->position);
228 if($oldTile !== null){
229 if($tileType === null || !($oldTile instanceof $tileType)){
230 $oldTile->close();
231 $oldTile = null;
232 }elseif($oldTile instanceof Spawnable){
233 $oldTile->clearSpawnCompoundCache(); //destroy old network cache
234 }
235 }
236 if($oldTile === null && $tileType !== null){
241 $tile = new $tileType($world, $this->position->asVector3());
242 $world->addTile($tile);
243 }

794 public function calculateIntercept(Vector3 $pos1, Vector3 $pos2) : ?RayTraceResult{
795 $bbs = $this->getCollisionBoxes();
796 if(count($bbs) === 0){
797 return null;
798 }
799
801 $currentHit = null;
803 $currentDistance = PHP_INT_MAX;
804
805 foreach($bbs as $bb){
806 $nextHit = $bb->calculateIntercept($pos1, $pos2);
807 if($nextHit === null){
808 continue;
809 }
810
811 $nextDistance = $nextHit->hitVector->distanceSquared($pos1);
812 if($nextDistance < $currentDistance){
813 $currentHit = $nextHit;
814 $currentDistance = $nextDistance;
815 }
816 }
817
818 return $currentHit;
819 }
