212 private function decodeFullState(int $data) : void{
213 $reader = new RuntimeDataReader($this->requiredBlockItemStateDataBits + $this->requiredBlockOnlyStateDataBits, $data);
214 $this->decodeBlockItemState($reader->readInt($this->requiredBlockItemStateDataBits));
215 $this->decodeBlockOnlyState($reader->readInt($this->requiredBlockOnlyStateDataBits));

218 private function encodeBlockItemState() : int{

220
221 $this->describeBlockItemState($writer);

224 throw new \LogicException(get_class($this) . ": Exactly $this->requiredBlockItemStateDataBits bits of block-item state data were expected, but $writtenBits were written");

242 private function encodeFullState() : int{
243 $writer = new RuntimeDataWriter($this->requiredBlockItemStateDataBits + $this->requiredBlockOnlyStateDataBits);
244 $writer->writeInt($this->requiredBlockItemStateDataBits, $this->encodeBlockItemState());
245 $writer->writeInt($this->requiredBlockOnlyStateDataBits, $this->encodeBlockOnlyState());
246
247 return $writer->getValue();

281 public function generateStatePermutations() : \Generator{
282 //TODO: this bruteforce approach to discovering all valid states is very inefficient for larger state data sizes
283 //at some point we'll need to find a better way to do this
284 $bits = $this->requiredBlockItemStateDataBits + $this->requiredBlockOnlyStateDataBits;
285 if($bits > Block::INTERNAL_STATE_DATA_BITS){
286 throw new \LogicException("Block state data cannot use more than " . Block::INTERNAL_STATE_DATA_BITS . " bits");
287 }
288 for($stateData = 0; $stateData < (1 << $bits); ++$stateData){
289 $v = clone $this;
290 try{
291 $v->decodeFullState($stateData);
292 if($v->encodeFullState() !== $stateData){
293 throw new \LogicException(static::class . "::decodeStateData() accepts invalid state data (returned " . $v->encodeFullState() . " for input $stateData)");
294 }
295 }catch(InvalidSerializedRuntimeDataException){ //invalid property combination, leave it
296 continue;
297 }
298
299 yield $v;
300 }

325 public function writeStateToWorld() : void{
326 $world = $this->position->getWorld();
327 $chunk = $world->getOrLoadChunkAtPosition($this->position);
328 if($chunk === null){
329 throw new AssumptionFailedError("World::setBlock() should have loaded the chunk before calling this method");
330 }
331 $chunk->setBlockStateId($this->position->x & Chunk::COORD_MASK, $this->position->y, $this->position->z & Chunk::COORD_MASK, $this->getStateId());
332
333 $tileType = $this->idInfo->getTileClass();
334 $oldTile = $world->getTile($this->position);
335 if($oldTile !== null){
336 if($tileType === null || !($oldTile instanceof $tileType)){
337 $oldTile->close();
338 $oldTile = null;
339 }elseif($oldTile instanceof Spawnable){
340 $oldTile->clearSpawnCompoundCache(); //destroy old network cache
341 }
342 }
343 if($oldTile === null && $tileType !== null){
348 $tile = new $tileType($world, $this->position->asVector3());
349 $world->addTile($tile);
350 }

860 public function calculateIntercept(Vector3 $pos1, Vector3 $pos2) : ?RayTraceResult{
861 $bbs = $this->getCollisionBoxes();
862 if(count($bbs) === 0){
863 return null;
864 }
865
867 $currentHit = null;
869 $currentDistance = PHP_INT_MAX;
870
871 foreach($bbs as $bb){
872 $nextHit = $bb->calculateIntercept($pos1, $pos2);
873 if($nextHit === null){
874 continue;
875 }
876
877 $nextDistance = $nextHit->hitVector->distanceSquared($pos1);
878 if($nextDistance < $currentDistance){
879 $currentHit = $nextHit;
880 $currentDistance = $nextDistance;
881 }
882 }
883
884 return $currentHit;
885 }
