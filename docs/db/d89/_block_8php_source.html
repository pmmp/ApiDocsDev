105 final public function decodeTypeData(int $data) : void{
106 $typeBits = $this->getRequiredTypeDataBits();
107 $givenBits = $typeBits;
108 $reader = new RuntimeDataReader($givenBits, $data);
109
110 $this->describeType($reader);
111 $readBits = $reader->getOffset();
112 if($typeBits !== $readBits){

120 final public function decodeStateData(int $data) : void{
121 $typeBits = $this->getRequiredTypeDataBits();
122 $stateBits = $this->getRequiredStateDataBits();
123 $givenBits = $typeBits + $stateBits;
124 $reader = new RuntimeDataReader($givenBits, $data);
125 $this->decodeTypeData($reader->readInt($typeBits));
126
127 $this->describeState($reader);
128 $readBits = $reader->getOffset() - $typeBits;
129 if($stateBits !== $readBits){
130 throw new \LogicException(get_class($this) . ": Exactly $stateBits bits of state data were provided, but $readBits were read");
131 }
132 }

137 final public function computeTypeData() : int{
138 $typeBits = $this->getRequiredTypeDataBits();
139 $requiredBits = $typeBits;
140 $writer = new RuntimeDataWriter($requiredBits);
141
142 $this->describeType($writer);
143 $writtenBits = $writer->getOffset();
144 if($typeBits !== $writtenBits){
145 throw new \LogicException(get_class($this) . ": Exactly $typeBits bits of type data were expected, but $writtenBits were written");
146 }
147
148 return $writer->getValue();
149 }

154 final public function computeStateData() : int{
155 $typeBits = $this->getRequiredTypeDataBits();
156 $stateBits = $this->getRequiredStateDataBits();
157 $requiredBits = $typeBits + $stateBits;
158 $writer = new RuntimeDataWriter($requiredBits);
159 $writer->int($typeBits, $this->computeTypeData());
160
161 $this->describeState($writer);
162 $writtenBits = $writer->getOffset() - $typeBits;
163 if($stateBits !== $writtenBits){
164 throw new \LogicException(get_class($this) . ": Exactly $stateBits bits of state data were expected, but $writtenBits were written");
165 }
166
167 return $writer->getValue();
168 }

194 public function writeStateToWorld() : void{
195 $world = $this->position->getWorld();
196 $world->getOrLoadChunkAtPosition($this->position)->setFullBlock($this->position->x & Chunk::COORD_MASK, $this->position->y, $this->position->z & Chunk::COORD_MASK, $this->getStateId());
197
198 $tileType = $this->idInfo->getTileClass();
199 $oldTile = $world->getTile($this->position);
200 if($oldTile !== null){
201 if($tileType === null || !($oldTile instanceof $tileType)){
202 $oldTile->close();
203 $oldTile = null;
204 }elseif($oldTile instanceof Spawnable){
205 $oldTile->clearSpawnCompoundCache(); //destroy old network cache
206 }
207 }
208 if($oldTile === null && $tileType !== null){
213 $tile = new $tileType($world, $this->position->asVector3());
214 $world->addTile($tile);

705 public function calculateIntercept(Vector3 $pos1, Vector3 $pos2) : ?RayTraceResult{
706 $bbs = $this->getCollisionBoxes();
707 if(count($bbs) === 0){
708 return null;
709 }
710
712 $currentHit = null;
714 $currentDistance = PHP_INT_MAX;
715
716 foreach($bbs as $bb){
717 $nextHit = $bb->calculateIntercept($pos1, $pos2);
718 if($nextHit === null){
719 continue;
720 }
721
722 $nextDistance = $nextHit->hitVector->distanceSquared($pos1);
723 if($nextDistance < $currentDistance){
724 $currentHit = $nextHit;
725 $currentDistance = $nextDistance;
726 }
727 }
728
729 return $currentHit;
730 }
