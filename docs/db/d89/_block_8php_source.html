122 final public function decodeTypeData(int $data) : void{
123 $typeBits = $this->getRequiredTypeDataBits();
124 $givenBits = $typeBits;
125 $reader = new RuntimeDataReader($givenBits, $data);
126
127 $this->describeType($reader);
128 $readBits = $reader->getOffset();
129 if($typeBits !== $readBits){

137 final public function decodeStateData(int $data) : void{
138 $typeBits = $this->getRequiredTypeDataBits();
139 $stateBits = $this->getRequiredStateDataBits();
140 $givenBits = $typeBits + $stateBits;
141 $reader = new RuntimeDataReader($givenBits, $data);
142 $this->decodeTypeData($reader->readInt($typeBits));
143
144 $this->describeState($reader);
145 $readBits = $reader->getOffset() - $typeBits;
146 if($stateBits !== $readBits){
147 throw new \LogicException(get_class($this) . ": Exactly $stateBits bits of state data were provided, but $readBits were read");
148 }
149 }

154 final public function computeTypeData() : int{
155 $typeBits = $this->getRequiredTypeDataBits();
156 $requiredBits = $typeBits;
157 $writer = new RuntimeDataWriter($requiredBits);
158
159 $this->describeType($writer);
160 $writtenBits = $writer->getOffset();
161 if($typeBits !== $writtenBits){
162 throw new \LogicException(get_class($this) . ": Exactly $typeBits bits of type data were expected, but $writtenBits were written");
163 }
164
165 return $writer->getValue();
166 }
167
171 final public function computeStateData() : int{
172 $typeBits = $this->getRequiredTypeDataBits();
173 $stateBits = $this->getRequiredStateDataBits();
174 $requiredBits = $typeBits + $stateBits;
175 $writer = new RuntimeDataWriter($requiredBits);
176 $writer->int($typeBits, $this->computeTypeData());
177
178 $this->describeState($writer);
179 $writtenBits = $writer->getOffset() - $typeBits;
180 if($stateBits !== $writtenBits){
181 throw new \LogicException(get_class($this) . ": Exactly $stateBits bits of state data were expected, but $writtenBits were written");
182 }
183
184 return $writer->getValue();
185 }

217 public function writeStateToWorld() : void{
218 $world = $this->position->getWorld();
219 $world->getOrLoadChunkAtPosition($this->position)->setFullBlock($this->position->x & Chunk::COORD_MASK, $this->position->y, $this->position->z & Chunk::COORD_MASK, $this->getStateId());
220
221 $tileType = $this->idInfo->getTileClass();
222 $oldTile = $world->getTile($this->position);
223 if($oldTile !== null){
224 if($tileType === null || !($oldTile instanceof $tileType)){
225 $oldTile->close();
226 $oldTile = null;
227 }elseif($oldTile instanceof Spawnable){
228 $oldTile->clearSpawnCompoundCache(); //destroy old network cache
229 }
230 }
231 if($oldTile === null && $tileType !== null){
236 $tile = new $tileType($world, $this->position->asVector3());
237 $world->addTile($tile);
238 }

781 public function calculateIntercept(Vector3 $pos1, Vector3 $pos2) : ?RayTraceResult{
782 $bbs = $this->getCollisionBoxes();
783 if(count($bbs) === 0){
784 return null;
785 }
786
788 $currentHit = null;
790 $currentDistance = PHP_INT_MAX;
791
792 foreach($bbs as $bb){
793 $nextHit = $bb->calculateIntercept($pos1, $pos2);
794 if($nextHit === null){
795 continue;
796 }
797
798 $nextDistance = $nextHit->hitVector->distanceSquared($pos1);
799 if($nextDistance < $currentDistance){
800 $currentHit = $nextHit;
801 $currentDistance = $nextDistance;
802 }
803 }
804
805 return $currentHit;
806 }
