150 public function getStateId() : int{
151 $typeId = $this->getTypeId();
152 //TODO: this XOR mask improves hashtable distribution, but it's only effective if the number of unique block
153 //type IDs is larger than the number of available state data bits. We should probably hash (e.g. using xxhash)
154 //the type ID to create a better mask.
155 //Alternatively, we could hash the whole state ID, but this is currently problematic, since we currently need
156 //to be able to recover the state data from the state ID because of UnknownBlock.
157 return ($typeId << self::INTERNAL_STATE_DATA_BITS) | ($this->encodeFullState() ^ ($typeId & self::INTERNAL_STATE_DATA_MASK));
158 }

207 private function decodeBlockItemState(int $data) : void{
208 $reader = new RuntimeDataReader($this->requiredBlockItemStateDataBits, $data);
209
210 $this->describeBlockItemState($reader);
211 $readBits = $reader->getOffset();
212 if($this->requiredBlockItemStateDataBits !== $readBits){
213 throw new \LogicException(get_class($this) . ": Exactly $this->requiredBlockItemStateDataBits bits of block-item state data were provided, but $readBits were read");

227 private function decodeFullState(int $data) : void{
228 $reader = new RuntimeDataReader($this->requiredBlockItemStateDataBits + $this->requiredBlockOnlyStateDataBits, $data);
229 $this->decodeBlockItemState($reader->readInt($this->requiredBlockItemStateDataBits));
230 $this->decodeBlockOnlyState($reader->readInt($this->requiredBlockOnlyStateDataBits));
231 }

233 private function encodeBlockItemState() : int{
234 $writer = new RuntimeDataWriter($this->requiredBlockItemStateDataBits);
235
236 $this->describeBlockItemState($writer);
237 $writtenBits = $writer->getOffset();
238 if($this->requiredBlockItemStateDataBits !== $writtenBits){
239 throw new \LogicException(get_class($this) . ": Exactly $this->requiredBlockItemStateDataBits bits of block-item state data were expected, but $writtenBits were written");
240 }
241
242 return $writer->getValue();
243 }

257 private function encodeFullState() : int{
258 $writer = new RuntimeDataWriter($this->requiredBlockItemStateDataBits + $this->requiredBlockOnlyStateDataBits);
259 $writer->writeInt($this->requiredBlockItemStateDataBits, $this->encodeBlockItemState());
260 $writer->writeInt($this->requiredBlockOnlyStateDataBits, $this->encodeBlockOnlyState());
261
262 return $writer->getValue();

296 public function generateStatePermutations() : \Generator{
297 //TODO: this bruteforce approach to discovering all valid states is very inefficient for larger state data sizes
298 //at some point we'll need to find a better way to do this
299 $bits = $this->requiredBlockItemStateDataBits + $this->requiredBlockOnlyStateDataBits;
300 if($bits > Block::INTERNAL_STATE_DATA_BITS){
301 throw new \LogicException("Block state data cannot use more than " . Block::INTERNAL_STATE_DATA_BITS . " bits");
302 }
303 for($stateData = 0; $stateData < (1 << $bits); ++$stateData){
304 $v = clone $this;
305 try{
306 $v->decodeFullState($stateData);
307 if($v->encodeFullState() !== $stateData){
308 throw new \LogicException(static::class . "::decodeStateData() accepts invalid state data (returned " . $v->encodeFullState() . " for input $stateData)");
309 }
310 }catch(InvalidSerializedRuntimeDataException){ //invalid property combination, leave it
311 continue;
312 }
313
314 yield $v;
315 }

340 public function writeStateToWorld() : void{
341 $world = $this->position->getWorld();
342 $chunk = $world->getOrLoadChunkAtPosition($this->position);
343 if($chunk === null){
344 throw new AssumptionFailedError("World::setBlock() should have loaded the chunk before calling this method");
345 }
346 $chunk->setBlockStateId($this->position->x & Chunk::COORD_MASK, $this->position->y, $this->position->z & Chunk::COORD_MASK, $this->getStateId());
347
348 $tileType = $this->idInfo->getTileClass();
349 $oldTile = $world->getTile($this->position);
350 if($oldTile !== null){
351 if($tileType === null || !($oldTile instanceof $tileType)){
352 $oldTile->close();
353 $oldTile = null;
354 }elseif($oldTile instanceof Spawnable){
355 $oldTile->clearSpawnCompoundCache(); //destroy old network cache
356 }
357 }
358 if($oldTile === null && $tileType !== null){
363 $tile = new $tileType($world, $this->position->asVector3());
364 $world->addTile($tile);
365 }

875 public function calculateIntercept(Vector3 $pos1, Vector3 $pos2) : ?RayTraceResult{
876 $bbs = $this->getCollisionBoxes();
877 if(count($bbs) === 0){
878 return null;
879 }
880
882 $currentHit = null;
884 $currentDistance = PHP_INT_MAX;
885
886 foreach($bbs as $bb){
887 $nextHit = $bb->calculateIntercept($pos1, $pos2);
888 if($nextHit === null){
889 continue;
890 }
891
892 $nextDistance = $nextHit->hitVector->distanceSquared($pos1);
893 if($nextDistance < $currentDistance){
894 $currentHit = $nextHit;
895 $currentDistance = $nextDistance;
896 }
897 }
898
899 return $currentHit;
900 }
