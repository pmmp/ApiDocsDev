189 private function decodeTypeData(int $data) : void{
190 $reader = new RuntimeDataReader($this->requiredTypeDataBits, $data);
191
192 $this->describeType($reader);
193 $readBits = $reader->getOffset();
194 if($this->requiredTypeDataBits !== $readBits){
195 throw new \LogicException(get_class($this) . ": Exactly $this->requiredTypeDataBits bits of type data were provided, but $readBits were read");
196 }
197 }

201
202 $this->describeState($reader);
203 $readBits = $reader->getOffset();
204 if($this->requiredStateDataBits !== $readBits){
205 throw new \LogicException(get_class($this) . ": Exactly $this->requiredStateDataBits bits of state data were provided, but $readBits were read");
206 }

209 private function decodeTypeAndStateData(int $data) : void{
210 $reader = new RuntimeDataReader($this->requiredTypeDataBits + $this->requiredStateDataBits, $data);
211 $this->decodeTypeData($reader->readInt($this->requiredTypeDataBits));
212 $this->decodeStateData($reader->readInt($this->requiredStateDataBits));
213 }

215 private function computeTypeData() : int{
216 $writer = new RuntimeDataWriter($this->requiredTypeDataBits);
217
218 $this->describeType($writer);
219 $writtenBits = $writer->getOffset();
220 if($this->requiredTypeDataBits !== $writtenBits){
221 throw new \LogicException(get_class($this) . ": Exactly $this->requiredTypeDataBits bits of type data were expected, but $writtenBits were written");
222 }
223
224 return $writer->getValue();
225 }

239 private function computeTypeAndStateData() : int{
240 $writer = new RuntimeDataWriter($this->requiredTypeDataBits + $this->requiredStateDataBits);
241 $writer->writeInt($this->requiredTypeDataBits, $this->computeTypeData());
242 $writer->writeInt($this->requiredStateDataBits, $this->computeStateData());
243
244 return $writer->getValue();

278 public function generateStatePermutations() : \Generator{
279 //TODO: this bruteforce approach to discovering all valid states is very inefficient for larger state data sizes
280 //at some point we'll need to find a better way to do this
281 $bits = $this->requiredTypeDataBits + $this->requiredStateDataBits;
282 if($bits > Block::INTERNAL_STATE_DATA_BITS){
283 throw new \LogicException("Block state data cannot use more than " . Block::INTERNAL_STATE_DATA_BITS . " bits");
284 }
285 for($stateData = 0; $stateData < (1 << $bits); ++$stateData){
286 $v = clone $this;
287 try{
288 $v->decodeTypeAndStateData($stateData);
289 if($v->computeTypeAndStateData() !== $stateData){
290 throw new \LogicException(static::class . "::decodeStateData() accepts invalid state data (returned " . $v->computeTypeAndStateData() . " for input $stateData)");
291 }
292 }catch(InvalidSerializedRuntimeDataException){ //invalid property combination, leave it
293 continue;
294 }
295
296 yield $v;
297 }

322 public function writeStateToWorld() : void{
323 $world = $this->position->getWorld();
324 $chunk = $world->getOrLoadChunkAtPosition($this->position);
325 if($chunk === null){
326 throw new AssumptionFailedError("World::setBlock() should have loaded the chunk before calling this method");
327 }
328 $chunk->setBlockStateId($this->position->x & Chunk::COORD_MASK, $this->position->y, $this->position->z & Chunk::COORD_MASK, $this->getStateId());
329
330 $tileType = $this->idInfo->getTileClass();
331 $oldTile = $world->getTile($this->position);
332 if($oldTile !== null){
333 if($tileType === null || !($oldTile instanceof $tileType)){
334 $oldTile->close();
335 $oldTile = null;
336 }elseif($oldTile instanceof Spawnable){
337 $oldTile->clearSpawnCompoundCache(); //destroy old network cache
338 }
339 }
340 if($oldTile === null && $tileType !== null){
345 $tile = new $tileType($world, $this->position->asVector3());
346 $world->addTile($tile);
347 }

857 public function calculateIntercept(Vector3 $pos1, Vector3 $pos2) : ?RayTraceResult{
858 $bbs = $this->getCollisionBoxes();
859 if(count($bbs) === 0){
860 return null;
861 }
862
864 $currentHit = null;
866 $currentDistance = PHP_INT_MAX;
867
868 foreach($bbs as $bb){
869 $nextHit = $bb->calculateIntercept($pos1, $pos2);
870 if($nextHit === null){
871 continue;
872 }
873
874 $nextDistance = $nextHit->hitVector->distanceSquared($pos1);
875 if($nextDistance < $currentDistance){
876 $currentHit = $nextHit;
877 $currentDistance = $nextDistance;
878 }
879 }
880
881 return $currentHit;
882 }
