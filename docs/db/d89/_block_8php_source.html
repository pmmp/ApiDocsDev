203 final public function decodeTypeData(int $data) : void{
204 $typeBits = $this->getRequiredTypeDataBits();
205 $reader = new RuntimeDataReader($typeBits, $data);
206
207 $this->describeType($reader);
208 $readBits = $reader->getOffset();
209 if($typeBits !== $readBits){
210 throw new \LogicException(get_class($this) . ": Exactly $typeBits bits of type data were provided, but $readBits were read");
211 }
212 }

217 final public function decodeTypeAndStateData(int $data) : void{
218 $typeBits = $this->getRequiredTypeDataBits();
219 $stateBits = $this->getRequiredStateDataBits();
220 $reader = new RuntimeDataReader($typeBits + $stateBits, $data);
221 $this->decodeTypeData($reader->readInt($typeBits));
222
223 $this->describeState($reader);
224 $readBits = $reader->getOffset() - $typeBits;
225 if($stateBits !== $readBits){
226 throw new \LogicException(get_class($this) . ": Exactly $stateBits bits of state data were provided, but $readBits were read");
227 }
228 }

233 final public function computeTypeData() : int{
234 $typeBits = $this->getRequiredTypeDataBits();
235 $writer = new RuntimeDataWriter($typeBits);
236
237 $this->describeType($writer);
238 $writtenBits = $writer->getOffset();
239 if($typeBits !== $writtenBits){
240 throw new \LogicException(get_class($this) . ": Exactly $typeBits bits of type data were expected, but $writtenBits were written");
241 }
242
243 return $writer->getValue();
244 }
245
249 final public function computeTypeAndStateData() : int{
250 $typeBits = $this->getRequiredTypeDataBits();
251 $stateBits = $this->getRequiredStateDataBits();
252 $writer = new RuntimeDataWriter($typeBits + $stateBits);
253 $writer->writeInt($typeBits, $this->computeTypeData());
254
255 $this->describeState($writer);
256 $writtenBits = $writer->getOffset() - $typeBits;
257 if($stateBits !== $writtenBits){
258 throw new \LogicException(get_class($this) . ": Exactly $stateBits bits of state data were expected, but $writtenBits were written");
259 }
260
261 return $writer->getValue();
262 }

310 public function writeStateToWorld() : void{
311 $world = $this->position->getWorld();
312 $chunk = $world->getOrLoadChunkAtPosition($this->position);
313 if($chunk === null){
314 throw new AssumptionFailedError("World::setBlock() should have loaded the chunk before calling this method");
315 }
316 $chunk->setBlockStateId($this->position->x & Chunk::COORD_MASK, $this->position->y, $this->position->z & Chunk::COORD_MASK, $this->getStateId());
317
318 $tileType = $this->idInfo->getTileClass();
319 $oldTile = $world->getTile($this->position);
320 if($oldTile !== null){
321 if($tileType === null || !($oldTile instanceof $tileType)){
322 $oldTile->close();
323 $oldTile = null;
324 }elseif($oldTile instanceof Spawnable){
325 $oldTile->clearSpawnCompoundCache(); //destroy old network cache
326 }
327 }
328 if($oldTile === null && $tileType !== null){
333 $tile = new $tileType($world, $this->position->asVector3());
334 $world->addTile($tile);
335 }

845 public function calculateIntercept(Vector3 $pos1, Vector3 $pos2) : ?RayTraceResult{
846 $bbs = $this->getCollisionBoxes();
847 if(count($bbs) === 0){
848 return null;
849 }
850
852 $currentHit = null;
854 $currentDistance = PHP_INT_MAX;
855
856 foreach($bbs as $bb){
857 $nextHit = $bb->calculateIntercept($pos1, $pos2);
858 if($nextHit === null){
859 continue;
860 }
861
862 $nextDistance = $nextHit->hitVector->distanceSquared($pos1);
863 if($nextDistance < $currentDistance){
864 $currentHit = $nextHit;
865 $currentDistance = $nextDistance;
866 }
867 }
868
869 return $currentHit;
870 }
