196 private function decodeTypeData(int $data) : void{
197 $reader = new RuntimeDataReader($this->requiredTypeDataBits, $data);
198
199 $this->describeType($reader);
200 $readBits = $reader->getOffset();
201 if($this->requiredTypeDataBits !== $readBits){

219 final public function decodeTypeAndStateData(int $data) : void{
220 $reader = new RuntimeDataReader($this->requiredTypeDataBits + $this->requiredStateDataBits, $data);
221 $this->decodeTypeData($reader->readInt($this->requiredTypeDataBits));
222 $this->decodeStateData($reader->readInt($this->requiredStateDataBits));
223 }

225 private function computeTypeData() : int{
226 $writer = new RuntimeDataWriter($this->requiredTypeDataBits);
227
228 $this->describeType($writer);
229 $writtenBits = $writer->getOffset();
230 if($this->requiredTypeDataBits !== $writtenBits){
231 throw new \LogicException(get_class($this) . ": Exactly $this->requiredTypeDataBits bits of type data were expected, but $writtenBits were written");
232 }
233
234 return $writer->getValue();
235 }

252 final public function computeTypeAndStateData() : int{
253 $writer = new RuntimeDataWriter($this->requiredTypeDataBits + $this->requiredStateDataBits);

306 public function writeStateToWorld() : void{
307 $world = $this->position->getWorld();
308 $chunk = $world->getOrLoadChunkAtPosition($this->position);
309 if($chunk === null){
310 throw new AssumptionFailedError("World::setBlock() should have loaded the chunk before calling this method");
311 }
312 $chunk->setBlockStateId($this->position->x & Chunk::COORD_MASK, $this->position->y, $this->position->z & Chunk::COORD_MASK, $this->getStateId());
313
314 $tileType = $this->idInfo->getTileClass();
315 $oldTile = $world->getTile($this->position);
316 if($oldTile !== null){
317 if($tileType === null || !($oldTile instanceof $tileType)){
318 $oldTile->close();
319 $oldTile = null;
320 }elseif($oldTile instanceof Spawnable){
321 $oldTile->clearSpawnCompoundCache(); //destroy old network cache
322 }
323 }
324 if($oldTile === null && $tileType !== null){
329 $tile = new $tileType($world, $this->position->asVector3());
330 $world->addTile($tile);
331 }

841 public function calculateIntercept(Vector3 $pos1, Vector3 $pos2) : ?RayTraceResult{
842 $bbs = $this->getCollisionBoxes();
843 if(count($bbs) === 0){
844 return null;
845 }
846
848 $currentHit = null;
850 $currentDistance = PHP_INT_MAX;
851
852 foreach($bbs as $bb){
853 $nextHit = $bb->calculateIntercept($pos1, $pos2);
854 if($nextHit === null){
855 continue;
856 }
857
858 $nextDistance = $nextHit->hitVector->distanceSquared($pos1);
859 if($nextDistance < $currentDistance){
860 $currentHit = $nextHit;
861 $currentDistance = $nextDistance;
862 }
863 }
864
865 return $currentHit;
866 }
