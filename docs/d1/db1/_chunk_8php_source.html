67 public function __construct(array $subChunks, BiomeArray $biomeIds, bool $terrainPopulated){
68 $this->subChunks = new \SplFixedArray(Chunk::MAX_SUBCHUNKS);
69
70 foreach($this->subChunks as $y => $null){
71 $this->subChunks[$y] = $subChunks[$y + self::MIN_SUBCHUNK_INDEX] ?? new SubChunk(BlockTypeIds::AIR << Block::INTERNAL_STATE_DATA_BITS, []);
72 }
73
74 $val = (self::MAX_SUBCHUNK_INDEX + 1) * SubChunk::EDGE_LENGTH;
75 $this->heightMap = HeightArray::fill($val); //TODO: what about lazily initializing this?
76 $this->biomeIds = $biomeIds;
77
78 $this->terrainPopulated = $terrainPopulated;

117 public function getHighestBlockAt(int $x, int $z) : ?int{
118 for($y = self::MAX_SUBCHUNK_INDEX; $y >= self::MIN_SUBCHUNK_INDEX; --$y){
119 $height = $this->getSubChunk($y)->getHighestBlockAt($x, $z);
120 if($height !== null){
121 return $height | ($y << SubChunk::COORD_BIT_SIZE);

189 public function addTile(Tile $tile) : void{
190 if($tile->isClosed()){
191 throw new \InvalidArgumentException("Attempted to add a garbage closed Tile to a chunk");
192 }
193
194 $pos = $tile->getPosition();
195 if(isset($this->tiles[$index = Chunk::blockHash($pos->x, $pos->y, $pos->z)]) && $this->tiles[$index] !== $tile){
196 throw new \InvalidArgumentException("Another tile is already at this location");
197 }
198 $this->tiles[$index] = $tile;
199 }

335 public static function blockHash(int $x, int $y, int $z) : int{
336 return ($y << (2 * SubChunk::COORD_BIT_SIZE)) |
337 (($z & SubChunk::COORD_MASK) << SubChunk::COORD_BIT_SIZE) |
338 ($x & SubChunk::COORD_MASK);
