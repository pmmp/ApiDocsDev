223 $users = $source->getServer()->getBroadcastChannelSubscribers(Server::BROADCAST_CHANNEL_ADMINISTRATIVE);
224 $result = KnownTranslationFactory::chat_type_admin($source->getName(), $message);
225 $colored = $result->prefix(TextFormat::GRAY . TextFormat::ITALIC);
226
227 if($sendToSource && !($source instanceof ConsoleCommandSender)){
228 $source->sendMessage($message);

231 foreach($users as $user){
