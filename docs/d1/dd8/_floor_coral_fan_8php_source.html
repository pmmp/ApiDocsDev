38final class FloorCoralFan extends BaseCoral{
39
40 protected BlockIdentifierFlattened $idInfoFlattened;
41
42 private int $axis = Axis::X;
43
44 public function __construct(BlockIdentifierFlattened $idInfo, string $name, BlockBreakInfo $breakInfo){
45 $this->idInfoFlattened = $idInfo;
46 parent::__construct($idInfo, $name, $breakInfo);
47 }
48
49 public function readStateFromData(int $id, int $stateMeta) : void{
50 $this->dead = $id === $this->idInfoFlattened->getSecondId();
51 $this->axis = ($stateMeta >> 3) === BlockLegacyMetadata::CORAL_FAN_EAST_WEST ? Axis::X : Axis::Z;
52 $coralType = CoralTypeIdMap::getInstance()->fromId($stateMeta & BlockLegacyMetadata::CORAL_FAN_TYPE_MASK);
53 if($coralType === null){
54 throw new InvalidBlockStateException("No such coral type");
55 }
56 $this->coralType = $coralType;
57 }
58
59 public function getId() : int{
60 return $this->dead ? $this->idInfoFlattened->getSecondId() : parent::getId();
61 }
62
63 public function asItem() : Item{
64 //TODO: HACK! workaround dead flag being lost when broken / blockpicked (original impl only uses first ID)
65 return ItemFactory::getInstance()->get(
66 $this->dead ? ItemIds::CORAL_FAN_DEAD : ItemIds::CORAL_FAN,
67 $this->writeStateToItemMeta()
68 );
69 }
70
71 protected function writeStateToMeta() : int{
72 return (($this->axis === Axis::X ? BlockLegacyMetadata::CORAL_FAN_EAST_WEST : BlockLegacyMetadata::CORAL_FAN_NORTH_SOUTH) << 3) |
73 CoralTypeIdMap::getInstance()->toId($this->coralType);
74 }
75
76 protected function writeStateToItemMeta() : int{
77 return CoralTypeIdMap::getInstance()->toId($this->coralType);
78 }
79
80 public function getStateBitmask() : int{
81 return 0b1111;
82 }
83
84 public function getAxis() : int{ return $this->axis; }
85
87 public function setAxis(int $axis) : self{
88 if($axis !== Axis::X && $axis !== Axis::Z){
89 throw new \InvalidArgumentException("Axis must be X or Z only");
90 }
91 $this->axis = $axis;
92 return $this;
93 }
94
95 public function place(BlockTransaction $tx, Item $item, Block $blockReplace, Block $blockClicked, int $face, Vector3 $clickVector, ?Player $player = null) : bool{
96 if(!$tx->fetchBlock($blockReplace->getPosition()->down())->isSolid()){
97 return false;
98 }
99 if($player !== null){
100 $playerBlockPos = $player->getPosition()->floor();
101 $directionVector = $blockReplace->getPosition()->subtractVector($playerBlockPos)->normalize();
102 $angle = rad2deg(atan2($directionVector->getZ(), $directionVector->getX()));
103
104 if($angle <= 45 || 315 <= $angle || (135 <= $angle && $angle <= 225)){
105 //TODO: This produces Z axis 75% of the time, because any negative angle will produce Z axis.
106 //This is a bug in vanilla. https://bugs.mojang.com/browse/MCPE-125311
107 $this->axis = Axis::Z;
108 }
109 }
110 return parent::place($tx, $item, $blockReplace, $blockClicked, $face, $clickVector, $player);
111 }
112
113 public function onNearbyBlockChange() : void{
114 $world = $this->position->getWorld();
115 if(!$world->getBlock($this->position->down())->isSolid()){
116 $world->useBreakOn($this->position);
117 }else{
118 parent::onNearbyBlockChange();
119 }
120 }
