77 private function getLeafTopOffset() : float{
78 return match($this->leafState){
79 DripleafState::STABLE(), DripleafState::UNSTABLE() => 1 / 16,
80 DripleafState::PARTIAL_TILT() => 3 / 16,
81 default => 0
82 };

85 public function onEntityInside(Entity $entity) : bool{
86 if(!$entity instanceof Projectile && $this->leafState->equals(DripleafState::STABLE())){
87 //the entity must be standing on top of the leaf - do not collapse if the entity is standing underneath
88 $intersection = AxisAlignedBB::one()
89 ->offset($this->position->x, $this->position->y, $this->position->z)
90 ->trim(Facing::DOWN, 1 - $this->getLeafTopOffset());
91 if($entity->getBoundingBox()->intersectsWith($intersection)){
92 $this->setTiltAndScheduleTick(DripleafState::UNSTABLE());
93 return false;
94 }
95 }
96 return true;
97 }
98
99 public function onProjectileHit(Projectile $projectile, RayTraceResult $hitResult) : void{
100 if(!$this->leafState->equals(DripleafState::FULL_TILT())){
101 $this->setTiltAndScheduleTick(DripleafState::FULL_TILT());
102 $this->position->getWorld()->addSound($this->position, new DripleafTiltDownSound());
103 }
104 }
105
106 public function onScheduledUpdate() : void{
107 if(!$this->leafState->equals(DripleafState::STABLE())){
108 if($this->leafState->equals(DripleafState::FULL_TILT())){
109 $this->position->getWorld()->setBlock($this->position, $this->setLeafState(DripleafState::STABLE()));
110 $this->position->getWorld()->addSound($this->position, new DripleafTiltUpSound());
111 }else{
112 $this->setTiltAndScheduleTick(match($this->leafState->id()){
113 DripleafState::UNSTABLE()->id() => DripleafState::PARTIAL_TILT(),
114 DripleafState::PARTIAL_TILT()->id() => DripleafState::FULL_TILT(),
115 default => throw new AssumptionFailedError("All types should be covered")
116 });
117 $this->position->getWorld()->addSound($this->position, new DripleafTiltDownSound());
118 }
119 }
120 }
121
122 protected function recalculateCollisionBoxes() : array{
123 if(!$this->leafState->equals(DripleafState::FULL_TILT())){
124 return [
125 AxisAlignedBB::one()
126 ->trim(Facing::DOWN, 11 / 16)
127 ->trim(Facing::UP, $this->getLeafTopOffset())
128 ];
129 }
130 return [];
131 }
