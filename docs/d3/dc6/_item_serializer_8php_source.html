95 public function serializeType(Item $item) : Data{
96 if($item->isNull()){
97 throw new \InvalidArgumentException("Cannot serialize a null itemstack");
98 }
99 if($item instanceof ItemBlock){
100 $data = $this->serializeBlockItem($item->getBlock());
101 }else{
102 $index = $item->getTypeId();
103
104 $locatedSerializer = $this->itemSerializers[$index][get_class($item)] ?? null;
105 if($locatedSerializer === null){
106 foreach(class_parents($item) as $parent){
107 if(isset($this->itemSerializers[$index][$parent])){
108 $locatedSerializer = $this->itemSerializers[$index][$parent];
109 break;
110 }
111 }
112 }
113
114 if($locatedSerializer === null){
115 throw new ItemTypeSerializeException("No serializer registered for " . get_class($item) . " ($index) " . $item->getName());
116 }
117
122 $serializer = $locatedSerializer;
123
125 $data = $serializer($item);
126 }
127
128 if($item->hasNamedTag()){
129 $resultTag = $item->getNamedTag();
130 $extraTag = $data->getTag();
131 if($extraTag !== null){
132 $resultTag = $resultTag->merge($extraTag);
133 }
134 $data = new Data($data->getName(), $data->getMeta(), $data->getBlock(), $resultTag);
135 }
136
137 return $data;
138 }

157 private function serializeBlockItem(Block $block) : Data{
158 $index = $block->getTypeId();
159
160 $locatedSerializer = $this->blockItemSerializers[$index][get_class($block)] ?? null;
161 if($locatedSerializer === null){
162 foreach(class_parents($block) as $parent){
163 if(isset($this->blockItemSerializers[$index][$parent])){
164 $locatedSerializer = $this->blockItemSerializers[$index][$parent];
165 break;
166 }
167 }
168 }
169
170 if($locatedSerializer !== null){
172 $serializer = $locatedSerializer;
173 $data = $serializer($block);
174 }else{
175 $data = $this->standardBlock($block);
176 }
177
178 return $data;

184 private function standardBlock(Block $block) : Data{
185 try{
186 $blockStateData = $this->blockStateSerializer->serialize($block->getStateId());

188 throw new ItemTypeSerializeException($e->getMessage(), 0, $e);
189 }
190
191 $itemNameId = BlockItemIdMap::getInstance()->lookupItemId($blockStateData->getName()) ?? $blockStateData->getName();
192
193 return new Data($itemNameId, 0, $blockStateData);
194 }
