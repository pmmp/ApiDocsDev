90 private static function generateAllStatesForType(Block $block) : \Generator{
91 //TODO: this bruteforce approach to discovering all valid states is very inefficient for larger state data sizes
92 //at some point we'll need to find a better way to do this
93 $bits = $block->getRequiredTypeDataBits() + $block->getRequiredStateDataBits();
94 if($bits > Block::INTERNAL_STATE_DATA_BITS){
95 throw new \InvalidArgumentException("Block state data cannot use more than " . Block::INTERNAL_STATE_DATA_BITS . " bits");
96 }
97 for($stateData = 0; $stateData < (1 << $bits); ++$stateData){
98 $v = clone $block;
99 try{
100 $v->decodeTypeAndStateData($stateData);
101 if($v->computeTypeAndStateData() !== $stateData){
102 //TODO: this should probably be a hard error
103 throw new \LogicException(get_class($block) . "::decodeStateData() accepts invalid state data (returned " . $v->computeTypeAndStateData() . " for input $stateData)");
104 }
105 }catch(InvalidSerializedRuntimeDataException){ //invalid property combination, leave it
106 continue;
107 }
108
109 yield $v;
110 }
111 }

125 public function register(Block $block, bool $override = false) : void{
126 $typeId = $block->getTypeId();
127
128 if(!$override && isset($this->typeIndex[$typeId])){
129 throw new \InvalidArgumentException("Block ID $typeId is already used by another block, and override was not requested");
130 }
131
132 $this->typeIndex[$typeId] = clone $block;
133
134 foreach(self::generateAllStatesForType($block) as $v){
135 $this->fillStaticArrays($v->getStateId(), $v);
136 }
137 }

139 private function fillStaticArrays(int $index, Block $block) : void{
140 $fullId = $block->getStateId();
141 if($index !== $fullId){
142 throw new AssumptionFailedError("Cannot fill static arrays for an invalid blockstate");
143 }else{
144 $this->fullList[$index] = $block;
145 $this->blastResistance[$index] = $block->getBreakInfo()->getBlastResistance();
146 $this->light[$index] = $block->getLightLevel();
147 $this->lightFilter[$index] = min(15, $block->getLightFilter() + LightUpdate::BASE_LIGHT_FILTER);
148 if($block->blocksDirectSkyLight()){
149 $this->blocksDirectSkyLight[$index] = true;
150 }
151 }
152 }

166 public function fromStateId(int $stateId) : Block{
167 if($stateId < 0){
168 throw new \InvalidArgumentException("Block state ID cannot be negative");
169 }
170 if(isset($this->fullList[$stateId])) { //hot
171 $block = clone $this->fullList[$stateId];
172 }else{
173 $typeId = $stateId >> Block::INTERNAL_STATE_DATA_BITS;
174 $stateData = $stateId & Block::INTERNAL_STATE_DATA_MASK;
175 $block = new UnknownBlock(new BID($typeId), new BlockTypeInfo(BreakInfo::instant()), $stateData);
176 }
177
178 return $block;
179 }
