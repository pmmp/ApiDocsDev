208 protected function storeLocal(string $key, mixed $complexData) : void{
209 if(self::$threadLocalStorage === null){
210 /*
211 * It's necessary to use an object (not array) here because pthreads is stupid. Non-default array statics
212 * will be inherited when task classes are copied to the worker thread, which would cause unwanted
213 * inheritance of primitive thread-locals, which we really don't want for various reasons.
214 * It won't try to inherit objects though, so this is the easiest solution.
215 */
216 self::$threadLocalStorage = new \ArrayObject();
217 }
218 self::$threadLocalStorage[spl_object_id($this)][$key] = $complexData;
