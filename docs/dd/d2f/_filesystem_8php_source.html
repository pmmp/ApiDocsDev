80 public static function recursiveUnlink(string $dir) : void{
81 if(is_dir($dir)){
82 $objects = Utils::assumeNotFalse(scandir($dir, SCANDIR_SORT_NONE), "scandir() shouldn't return false when is_dir() returns true");
83 foreach($objects as $object){
84 if($object !== "." && $object !== ".."){
85 $fullObject = Path::join($dir, $object);
86 if(is_dir($fullObject)){
87 self::recursiveUnlink($fullObject);
88 }else{
89 unlink($fullObject);
90 }
91 }
92 }
93 rmdir($dir);
94 }elseif(is_file($dir)){
95 unlink($dir);
96 }

102 public static function recursiveCopy(string $origin, string $destination) : void{
103 if(!is_dir($origin)){
104 throw new \RuntimeException("$origin does not exist, or is not a directory");
105 }
106 if(!is_dir($destination)){
107 if(file_exists($destination)){
108 throw new \RuntimeException("$destination already exists, and is not a directory");
109 }
110 if(!is_dir(dirname($destination))){
111 //if the parent dir doesn't exist, the user most likely made a mistake
112 throw new \RuntimeException("The parent directory of $destination does not exist, or is not a directory");
113 }
114 try{
115 ErrorToExceptionHandler::trap(fn() => mkdir($destination));
116 }catch(\ErrorException $e){
117 if(!is_dir($destination)){
118 throw new \RuntimeException("Failed to create output directory $destination: " . $e->getMessage());
119 }
120 }
121 }
122 self::recursiveCopyInternal($origin, $destination);
123 }

125 private static function recursiveCopyInternal(string $origin, string $destination) : void{
126 if(is_dir($origin)){
127 if(!is_dir($destination)){
128 if(file_exists($destination)){
129 throw new \RuntimeException("Path $destination does not exist, or is not a directory");
130 }
131 mkdir($destination); //TODO: access permissions?
132 }
133 $objects = Utils::assumeNotFalse(scandir($origin, SCANDIR_SORT_NONE));
134 foreach($objects as $object){
135 if($object === "." || $object === ".."){
136 continue;
137 }
138 self::recursiveCopyInternal(Path::join($origin, $object), Path::join($destination, $object));
139 }
140 }else{
141 $dirName = dirname($destination);
142 if(!is_dir($dirName)){ //the destination folder should already exist
143 throw new AssumptionFailedError("The destination folder should have been created in the parent call");
144 }
145 copy($origin, $destination);
146 }

167 public static function cleanPath($path){
168 $result = str_replace([DIRECTORY_SEPARATOR, ".php", "phar://"], ["/", "", ""], $path);
169
170 //remove relative paths
171 //this should probably never have integer keys, but it's safer than making PHPStan ignore it
172 foreach(Utils::stringifyKeys(self::$cleanedPaths) as $cleanPath => $replacement){
173 $cleanPath = rtrim(str_replace([DIRECTORY_SEPARATOR, "phar://"], ["/", ""], $cleanPath), "/");
174 if(strpos($result, $cleanPath) === 0){
175 $result = ltrim(str_replace($cleanPath, $replacement, $result), "/");
176 }
177 }
178 return $result;
179 }

189 public static function createLockFile(string $lockFilePath) : ?int{
190 $resource = fopen($lockFilePath, "a+b");
191 if($resource === false){
192 throw new \InvalidArgumentException("Invalid lock file path or read/write permissions denied");
193 }
194 if(!flock($resource, LOCK_EX | LOCK_NB)){
195 //wait for a shared lock to avoid race conditions if two servers started at the same time - this makes sure the
196 //other server wrote its PID and released exclusive lock before we get our lock
197 flock($resource, LOCK_SH);
198 $pid = Utils::assumeNotFalse(stream_get_contents($resource), "This is a known valid file resource, at worst we should receive an empty string");
199 if(preg_match('/^\d+$/', $pid) === 1){
200 return (int) $pid;
201 }
202 return -1;
203 }
204 ftruncate($resource, 0);
205 fwrite($resource, (string) getmypid());
206 fflush($resource);
207 flock($resource, LOCK_SH); //prevent acquiring an exclusive lock from another process, but allow reading
208 self::$lockFileHandles[realpath($lockFilePath)] = $resource; //keep the resource alive to preserve the lock
209 return null;
210 }

217 public static function releaseLockFile(string $lockFilePath) : void{
218 $lockFilePath = realpath($lockFilePath);
219 if($lockFilePath === false){
220 throw new \InvalidArgumentException("Invalid lock file path");
221 }
222 if(isset(self::$lockFileHandles[$lockFilePath])){
223 flock(self::$lockFileHandles[$lockFilePath], LOCK_UN);
224 fclose(self::$lockFileHandles[$lockFilePath]);
225 unset(self::$lockFileHandles[$lockFilePath]);
226 @unlink($lockFilePath);
227 }

241 public static function safeFilePutContents(string $fileName, string $contents, int $flags = 0, $context = null) : void{
242 $directory = dirname($fileName);
243 if(!is_dir($directory)){
244 throw new \RuntimeException("Target directory path does not exist or is not a directory");
245 }
246 if(is_dir($fileName)){
247 throw new \RuntimeException("Target file path already exists and is not a file");
248 }
249
250 $counter = 0;
251 do{
252 //we don't care about overwriting any preexisting tmpfile but we can't write if a directory is already here
253 $temporaryFileName = $fileName . ".$counter.tmp";
254 $counter++;
255 }while(is_dir($temporaryFileName));
256
257 try{
258 ErrorToExceptionHandler::trap(fn() => $context !== null ?
259 file_put_contents($temporaryFileName, $contents, $flags, $context) :
260 file_put_contents($temporaryFileName, $contents, $flags)
261 );
262 }catch(\ErrorException $filePutContentsException){
263 $context !== null ?
264 @unlink($temporaryFileName, $context) :
265 @unlink($temporaryFileName);
266 throw new \RuntimeException("Failed to write to temporary file $temporaryFileName: " . $filePutContentsException->getMessage(), 0, $filePutContentsException);
267 }
268
269 $renameTemporaryFileResult = $context !== null ?
270 @rename($temporaryFileName, $fileName, $context) :
271 @rename($temporaryFileName, $fileName);
272 if(!$renameTemporaryFileResult){
273 /*
274 * The following code works around a bug in Windows where rename() will periodically decide to give us a
275 * spurious "Access is denied (code: 5)" error. As far as I could determine, the fault comes from Windows
276 * itself, but since I couldn't reliably reproduce the issue it's very hard to debug.
277 *
278 * The following code can be used to test. Usually it will fail anywhere before 100,000 iterations.
279 *
280 * for($i = 0; $i < 10_000_000; ++$i){
281 * file_put_contents('ops.txt.0.tmp', 'some data ' . $i, 0);
282 * if(!rename('ops.txt.0.tmp', 'ops.txt')){
283 * throw new \Error("something weird happened");
284 * }
285 * }
286 */
287 try{
288 ErrorToExceptionHandler::trap(fn() => $context !== null ?
289 copy($temporaryFileName, $fileName, $context) :
290 copy($temporaryFileName, $fileName)
291 );
292 }catch(\ErrorException $copyException){
293 throw new \RuntimeException("Failed to move temporary file contents into target file: " . $copyException->getMessage(), 0, $copyException);
294 }
295 @unlink($temporaryFileName);
296 }
