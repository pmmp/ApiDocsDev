121 private static function deduplicateCompound(CompoundTag $tag, array &$keyIndex, array &$valueIndex) : CompoundTag{
122 if($tag->count() === 0){
123 return $tag;
124 }
125
126 $newTag = CompoundTag::create();
127 foreach($tag as $key => $value){
128 $key = $keyIndex[$key] ??= $key;
129
130 if($value instanceof CompoundTag){
131 $value = self::deduplicateCompound($value, $keyIndex, $valueIndex);
132 }elseif($value instanceof ByteTag || $value instanceof IntTag || $value instanceof StringTag){
133 $value = $valueIndex[$value->getType()][$value->getValue()] ??= $value;
134 }
135
136 $newTag->setTag($key, $value);
137 }
138
139 return $newTag;
140 }

160 public static function loadFromString(string $blockPaletteContents, string $metaMapContents) : self{
161 $metaMap = json_decode($metaMapContents, flags: JSON_THROW_ON_ERROR);
162 if(!is_array($metaMap)){
163 throw new \InvalidArgumentException("Invalid metaMap, expected array for root type, got " . get_debug_type($metaMap));
164 }
165
166 $entries = [];
167
168 foreach(self::loadPaletteFromString($blockPaletteContents) as $i => $state){
169 $meta = $metaMap[$i] ?? null;
170 if($meta === null){
171 throw new \InvalidArgumentException("Missing associated meta value for state $i (" . $state->toNbt() . ")");
172 }
173 if(!is_int($meta)){
174 throw new \InvalidArgumentException("Invalid metaMap offset $i, expected int, got " . get_debug_type($meta));
175 }
176 $entries[$i] = new BlockStateDictionaryEntry($state, $meta);
177 }
178
179 return new self($entries);
