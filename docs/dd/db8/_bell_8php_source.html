117 public function place(BlockTransaction $tx, Item $item, Block $blockReplace, Block $blockClicked, int $face, Vector3 $clickVector, ?Player $player = null) : bool{
118 if($face === Facing::UP){
119 if(!$this->canBeSupportedBy($tx->fetchBlock($this->position->down()))){
120 return false;
121 }
122 if($player !== null){
123 $this->setFacing(Facing::opposite($player->getHorizontalFacing()));
124 }
125 $this->setAttachmentType(BellAttachmentType::FLOOR());
126 }elseif($face === Facing::DOWN){
127 if(!$this->canBeSupportedBy($tx->fetchBlock($this->position->up()))){
128 return false;
129 }
130 $this->setAttachmentType(BellAttachmentType::CEILING());
131 }else{
132 $this->setFacing($face);
133 if($this->canBeSupportedBy($tx->fetchBlock($this->position->getSide(Facing::opposite($face))))){
134 $this->setAttachmentType(BellAttachmentType::ONE_WALL());
135 }else{
136 return false;
137 }
138 if($this->canBeSupportedBy($tx->fetchBlock($this->position->getSide($face)))){
139 $this->setAttachmentType(BellAttachmentType::TWO_WALLS());
140 }
141 }

151 ){

156 public function onInteract(Item $item, int $face, Vector3 $clickVector, ?Player $player = null) : bool{
157 if($player !== null){
158 $faceHit = Facing::opposite($player->getHorizontalFacing());
159 if($this->attachmentType->equals(BellAttachmentType::CEILING())){
160 $this->ring($faceHit);
161 }
162 if($this->attachmentType->equals(BellAttachmentType::FLOOR()) && Facing::axis($faceHit) === Facing::axis($this->facing)){
163 $this->ring($faceHit);
164 }
165 if(
166 ($this->attachmentType->equals(BellAttachmentType::ONE_WALL()) || $this->attachmentType->equals(BellAttachmentType::TWO_WALLS())) &&
167 ($faceHit === Facing::rotateY($this->facing, false) || $faceHit === Facing::rotateY($this->facing, true))
168 ){
169 $this->ring($faceHit);
170 }
171 }
172
173 return true;
174 }

178 $tile = $this->position->getWorld()->getTile($this->position);
179 if($tile instanceof TileBell){
180 $this->position->getWorld()->broadcastPacketToViewers($this->position, $tile->createFakeUpdatePacket($faceHit));
181 }
