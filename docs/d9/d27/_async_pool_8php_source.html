135 private function getWorker(int $worker) : AsyncWorker{
136 if(!isset($this->workers[$worker])){
137 $notifier = new SleeperNotifier();
138 $this->workers[$worker] = new AsyncWorker($this->logger, $worker, $this->workerMemoryLimit, $notifier);
139 $this->eventLoop->addNotifier($notifier, function() use ($worker) : void{
140 $this->collectTasksFromWorker($worker);
141 });
142 $this->workers[$worker]->setClassLoaders([$this->classLoader]);
143 $this->workers[$worker]->start(self::WORKER_START_OPTIONS);
144
145 $this->taskQueues[$worker] = new \SplQueue();
146
147 foreach($this->workerStartHooks as $hook){
148 $hook($worker);
149 }
150 }
151
152 return $this->workers[$worker];
153 }

158 public function submitTaskToWorker(AsyncTask $task, int $worker) : void{
159 if($worker < 0 || $worker >= $this->size){
160 throw new \InvalidArgumentException("Invalid worker $worker");
161 }
162 if($task->isSubmitted()){
163 throw new \InvalidArgumentException("Cannot submit the same AsyncTask instance more than once");
164 }
165
166 $task->progressUpdates = new \Threaded;
167 $task->setSubmitted();
168
169 $this->getWorker($worker)->stack($task);
170 $this->taskQueues[$worker]->enqueue($task);
171 $this->workerLastUsed[$worker] = time();
172 }

181 public function selectWorker() : int{
182 $worker = null;
183 $minUsage = PHP_INT_MAX;
184 foreach($this->taskQueues as $i => $queue){
185 if(($usage = $queue->count()) < $minUsage){
186 $worker = $i;
187 $minUsage = $usage;
188 if($usage === 0){
189 break;
190 }
191 }
192 }
193 if($worker === null || ($minUsage > 0 && count($this->workers) < $this->size)){
194 //select a worker to start on the fly
195 for($i = 0; $i < $this->size; ++$i){
196 if(!isset($this->workers[$i])){
197 $worker = $i;
198 break;
199 }
200 }
201 }
202
203 assert($worker !== null);
204 return $worker;
205 }

241 public function collectTasksFromWorker(int $worker) : bool{
242 if(!isset($this->taskQueues[$worker])){
243 throw new \InvalidArgumentException("No such worker $worker");
244 }
245 $queue = $this->taskQueues[$worker];
246 $more = false;
247 while(!$queue->isEmpty()){
249 $task = $queue->bottom();
250 $task->checkProgressUpdates();
251 if($task->isFinished()){ //make sure the task actually executed before trying to collect
252 $queue->dequeue();
253
254 if($task->isCrashed()){
255 $this->logger->critical("Could not execute asynchronous task " . (new \ReflectionClass($task))->getShortName() . ": Task crashed");
256 $task->onError();
257 }elseif(!$task->hasCancelledRun()){
258 /*
259 * It's possible for a task to submit a progress update and then finish before the progress
260 * update is detected by the parent thread, so here we consume any missed updates.
261 *
262 * When this happens, it's possible for a progress update to arrive between the previous
263 * checkProgressUpdates() call and the next isGarbage() call, causing progress updates to be
264 * lost. Thus, it's necessary to do one last check here to make sure all progress updates have
265 * been consumed before completing.
266 */
267 $task->checkProgressUpdates();
268 $task->onCompletion();
269 }
270 }else{
271 $more = true;
272 break; //current task is still running, skip to next worker
273 }
274 }
275 $this->workers[$worker]->collect();
276 return $more;
277 }

289 public function shutdownUnusedWorkers() : int{
290 $ret = 0;
291 $time = time();
292 foreach($this->taskQueues as $i => $queue){
293 if((!isset($this->workerLastUsed[$i]) || $this->workerLastUsed[$i] + 300 < $time) && $queue->isEmpty()){
294 $this->workers[$i]->quit();
295 $this->eventLoop->removeNotifier($this->workers[$i]->getNotifier());
296 unset($this->workers[$i], $this->taskQueues[$i], $this->workerLastUsed[$i]);
297 $ret++;
298 }
299 }
300
301 return $ret;
302 }
