130 private function getWorker(int $worker) : AsyncWorker{
131 if(!isset($this->workers[$worker])){
132 $notifier = new SleeperNotifier();
133 $this->workers[$worker] = new AsyncWorker($this->logger, $worker, $this->workerMemoryLimit, $notifier);
134 $this->eventLoop->addNotifier($notifier, function() use ($worker) : void{
135 $this->collectTasksFromWorker($worker);
136 });
137 $this->workers[$worker]->setClassLoaders([$this->classLoader]);
138 $this->workers[$worker]->start(self::WORKER_START_OPTIONS);
139
140 $this->taskQueues[$worker] = new \SplQueue();
141
142 foreach($this->workerStartHooks as $hook){
143 $hook($worker);
144 }
145 }
146
147 return $this->workers[$worker];
148 }

153 public function submitTaskToWorker(AsyncTask $task, int $worker) : void{
154 if($worker < 0 || $worker >= $this->size){
155 throw new \InvalidArgumentException("Invalid worker $worker");
156 }
157 if($task->isSubmitted()){
158 throw new \InvalidArgumentException("Cannot submit the same AsyncTask instance more than once");
159 }
160
161 $task->progressUpdates = new \Threaded;
162 $task->setSubmitted();
163
164 $this->getWorker($worker)->stack($task);
165 $this->taskQueues[$worker]->enqueue($task);
166 $this->workerLastUsed[$worker] = time();
167 }

176 public function selectWorker() : int{
177 $worker = null;
178 $minUsage = PHP_INT_MAX;
179 foreach($this->taskQueues as $i => $queue){
180 if(($usage = $queue->count()) < $minUsage){
181 $worker = $i;
182 $minUsage = $usage;
183 if($usage === 0){
184 break;
185 }
186 }
187 }
188 if($worker === null || ($minUsage > 0 && count($this->workers) < $this->size)){
189 //select a worker to start on the fly
190 for($i = 0; $i < $this->size; ++$i){
191 if(!isset($this->workers[$i])){
192 $worker = $i;
193 break;
194 }
195 }
196 }
197
198 assert($worker !== null);
199 return $worker;
200 }

236 public function collectTasksFromWorker(int $worker) : bool{
237 if(!isset($this->taskQueues[$worker])){
238 throw new \InvalidArgumentException("No such worker $worker");
239 }
240 $queue = $this->taskQueues[$worker];
241 $more = false;
242 while(!$queue->isEmpty()){
244 $task = $queue->bottom();
245 $task->checkProgressUpdates();
246 if($task->isFinished()){ //make sure the task actually executed before trying to collect
247 $queue->dequeue();
248
249 if($task->isCrashed()){
250 $this->logger->critical("Could not execute asynchronous task " . (new \ReflectionClass($task))->getShortName() . ": Task crashed");
251 $task->onError();
252 }elseif(!$task->hasCancelledRun()){
253 /*
254 * It's possible for a task to submit a progress update and then finish before the progress
255 * update is detected by the parent thread, so here we consume any missed updates.
256 *
257 * When this happens, it's possible for a progress update to arrive between the previous
258 * checkProgressUpdates() call and the next isGarbage() call, causing progress updates to be
259 * lost. Thus, it's necessary to do one last check here to make sure all progress updates have
260 * been consumed before completing.
261 */
262 $task->checkProgressUpdates();
263 $task->onCompletion();
264 }
265 }else{
266 $more = true;
267 break; //current task is still running, skip to next worker
268 }
269 }
270 $this->workers[$worker]->collect();
271 return $more;
272 }

284 public function shutdownUnusedWorkers() : int{
285 $ret = 0;
286 $time = time();
287 foreach($this->taskQueues as $i => $queue){
288 if((!isset($this->workerLastUsed[$i]) || $this->workerLastUsed[$i] + 300 < $time) && $queue->isEmpty()){
289 $this->workers[$i]->quit();
290 $this->eventLoop->removeNotifier($this->workers[$i]->getNotifier());
291 unset($this->workers[$i], $this->taskQueues[$i], $this->workerLastUsed[$i]);
292 $ret++;
293 }
294 }
295
296 return $ret;
297 }
