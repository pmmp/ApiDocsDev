121 private function getWorker(int $workerId) : AsyncPoolWorkerEntry{
122 if(!isset($this->workers[$workerId])){
123 $sleeperEntry = $this->eventLoop->addNotifier(function() use ($workerId) : void{
124 $this->collectTasksFromWorker($workerId);
125 });
126 $this->workers[$workerId] = new AsyncPoolWorkerEntry(new AsyncWorker($this->logger, $workerId, $this->workerMemoryLimit, $sleeperEntry), $sleeperEntry->getNotifierId());
127 $this->workers[$workerId]->worker->setClassLoaders([$this->classLoader]);
128 $this->workers[$workerId]->worker->start(self::WORKER_START_OPTIONS);
129
130 foreach($this->workerStartHooks as $hook){
131 $hook($workerId);
132 }
133 }
134
135 return $this->workers[$workerId];

162 public function selectWorker() : int{
163 $worker = null;
164 $minUsage = PHP_INT_MAX;
165 foreach($this->workers as $i => $entry){
166 if(($usage = $entry->tasks->count()) < $minUsage){
167 $worker = $i;
168 $minUsage = $usage;
169 if($usage === 0){
170 break;
171 }
172 }
173 }
174 if($worker === null || ($minUsage > 0 && count($this->workers) < $this->size)){
175 //select a worker to start on the fly
176 for($i = 0; $i < $this->size; ++$i){
177 if(!isset($this->workers[$i])){
178 $worker = $i;
179 break;
180 }
181 }
182 }
183
184 assert($worker !== null);
185 return $worker;
186 }

208 public function collectTasks() : bool{

222 public function collectTasksFromWorker(int $worker) : bool{
223 if(!isset($this->workers[$worker])){
224 throw new \InvalidArgumentException("No such worker $worker");
225 }
226 $queue = $this->workers[$worker]->tasks;
227 $more = false;
228 while(!$queue->isEmpty()){
230 $task = $queue->bottom();
231 if($task->isFinished()){ //make sure the task actually executed before trying to collect
232 $queue->dequeue();
233
234 if($task->isCrashed()){
235 $this->logger->critical("Could not execute asynchronous task " . (new \ReflectionClass($task))->getShortName() . ": Task crashed");
236 $task->onError();
237 }elseif(!$task->hasCancelledRun()){
238 /*
239 * It's possible for a task to submit a progress update and then finish before the progress
240 * update is detected by the parent thread, so here we consume any missed updates.
241 *
242 * When this happens, it's possible for a progress update to arrive between the previous
243 * checkProgressUpdates() call and the next isGarbage() call, causing progress updates to be
244 * lost. Thus, it's necessary to do one last check here to make sure all progress updates have
245 * been consumed before completing.
246 */
247 $task->checkProgressUpdates();
248 $task->onCompletion();
249 }
250 }else{
251 $task->checkProgressUpdates();
252 $more = true;
253 break; //current task is still running, skip to next worker
254 }
255 }
256 $this->workers[$worker]->worker->collect();
257 return $more;
258 }

270 public function shutdownUnusedWorkers() : int{
271 $ret = 0;
272 $time = time();
273 foreach($this->workers as $i => $entry){
274 if($entry->lastUsed + 300 < $time && $entry->tasks->isEmpty()){
275 $entry->worker->quit();
276 $this->eventLoop->removeNotifier($entry->sleeperNotifierId);
277 unset($this->workers[$i]);
278 $ret++;
279 }
