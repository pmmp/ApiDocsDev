64 public static function parseBlockNBT(BlockFactory $factory, CompoundTag $nbt) : Block{
65 $blockId = 0;
66
67 //TODO: 1.8+ save format
68 if(($tileIdTag = $nbt->getTag("TileID")) instanceof IntTag){
69 $blockId = $tileIdTag->getValue();
70 }elseif(($tileTag = $nbt->getTag("Tile")) instanceof ByteTag){
71 $blockId = $tileTag->getValue();
72 }
73
74 if($blockId === 0){
75 throw new SavedDataLoadingException("Missing block info from NBT");
76 }
77
78 $damage = $nbt->getByte("Data", 0);

97 protected function entityBaseTick(int $tickDiff = 1) : bool{
98 if($this->closed){
99 return false;
100 }
101
102 $hasUpdate = parent::entityBaseTick($tickDiff);
103
104 if(!$this->isFlaggedForDespawn()){
105 $world = $this->getWorld();
106 $pos = $this->location->add(-$this->size->getWidth() / 2, $this->size->getHeight(), -$this->size->getWidth() / 2)->floor();
107
108 $this->block->position($world, $pos->x, $pos->y, $pos->z);
109
110 $blockTarget = null;
111 if($this->block instanceof Fallable){
112 $blockTarget = $this->block->tickFalling();
113 }
114
115 if($this->onGround || $blockTarget !== null){
116 $this->flagForDespawn();
117
118 $block = $world->getBlock($pos);
119 if(!$block->canBeReplaced() || !$world->isInWorld($pos->getFloorX(), $pos->getFloorY(), $pos->getFloorZ()) || ($this->onGround && abs($this->location->y - $this->location->getFloorY()) > 0.001)){
120 //FIXME: anvils are supposed to destroy torches
121 $world->dropItem($this->location, $this->block->asItem());
122 }else{
123 $ev = new EntityBlockChangeEvent($this, $block, $blockTarget ?? $this->block);
124 $ev->call();
125 if(!$ev->isCancelled()){
126 $world->setBlock($pos, $ev->getTo());
127 }
128 }
129 $hasUpdate = true;
130 }
131 }
132
133 return $hasUpdate;
134 }
