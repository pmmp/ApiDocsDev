40final class RuntimeBlockMapping{
41 use SingletonTrait;
42
47 private array $networkIdCache = [];
48
50 private BlockStateData $fallbackStateData;
51 private int $fallbackStateId;
52
53 private static function make() : self{
54 $canonicalBlockStatesFile = Path::join(\pocketmine\BEDROCK_DATA_PATH, "canonical_block_states.nbt");
55 $canonicalBlockStatesRaw = Utils::assumeNotFalse(file_get_contents($canonicalBlockStatesFile), "Missing required resource file");
56
57 $metaMappingFile = Path::join(\pocketmine\BEDROCK_DATA_PATH, 'block_state_meta_map.json');
58 $metaMappingRaw = Utils::assumeNotFalse(file_get_contents($metaMappingFile), "Missing required resource file");
59 return new self(
60 BlockStateDictionary::loadFromString($canonicalBlockStatesRaw, $metaMappingRaw),
61 GlobalBlockStateHandlers::getSerializer()
62 );
63 }
64
65 public function __construct(
66 private BlockStateDictionary $blockStateDictionary,
67 private BlockStateSerializer $blockStateSerializer
68 ){
69 $this->fallbackStateId = $this->blockStateDictionary->lookupStateIdFromData(
70 new BlockStateData(BlockTypeNames::INFO_UPDATE, [], BlockStateData::CURRENT_VERSION)
71 ) ?? throw new AssumptionFailedError(BlockTypeNames::INFO_UPDATE . " should always exist");
72 //lookup the state data from the dictionary to avoid keeping two copies of the same data around
73 $this->fallbackStateData = $this->blockStateDictionary->getDataFromStateId($this->fallbackStateId) ?? throw new AssumptionFailedError("We just looked up this state data, so it must exist");
74 }
75
76 public function toRuntimeId(int $internalStateId) : int{
77 if(isset($this->networkIdCache[$internalStateId])){
78 return $this->networkIdCache[$internalStateId];
79 }
80
81 try{
82 $blockStateData = $this->blockStateSerializer->serialize($internalStateId);
83
84 $networkId = $this->blockStateDictionary->lookupStateIdFromData($blockStateData);
85 if($networkId === null){
86 throw new AssumptionFailedError("Unmapped blockstate returned by blockstate serializer: " . $blockStateData->toNbt());
87 }
88 }catch(BlockStateSerializeException){
89 //TODO: this will swallow any error caused by invalid block properties; this is not ideal, but it should be
90 //covered by unit tests, so this is probably a safe assumption.
91 $networkId = $this->fallbackStateId;
92 }
93
94 return $this->networkIdCache[$internalStateId] = $networkId;
95 }
96
100 public function toStateData(int $internalStateId) : BlockStateData{
101 //we don't directly use the blockstate serializer here - we can't assume that the network blockstate NBT is the
102 //same as the disk blockstate NBT, in case we decide to have different world version than network version (or in
103 //case someone wants to implement multi version).
104 $networkRuntimeId = $this->toRuntimeId($internalStateId);
105
106 return $this->blockStateDictionary->getDataFromStateId($networkRuntimeId) ?? throw new AssumptionFailedError("We just looked up this state ID, so it must exist");
107 }
108
109 public function getBlockStateDictionary() : BlockStateDictionary{ return $this->blockStateDictionary; }
110
111 public function getFallbackStateData() : BlockStateData{ return $this->fallbackStateData; }
