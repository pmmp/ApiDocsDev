51 private function __construct(){
52 $stream = PacketSerializer::decoder(
53 Utils::assumeNotFalse(file_get_contents(Path::join(\pocketmine\BEDROCK_DATA_PATH, "canonical_block_states.nbt")), "Missing required resource file"),
54 0,
55 new PacketSerializerContext(GlobalItemTypeDictionary::getInstance()->getDictionary())
56 );
57 $list = [];
58 while(!$stream->feof()){
59 $list[] = $stream->getNbtCompoundRoot();
60 }
61 $this->bedrockKnownStates = $list;
62
63 $this->setupLegacyMappings();
64 }

66 private function setupLegacyMappings() : void{
67 $legacyIdMap = LegacyBlockIdToStringIdMap::getInstance();
69 $legacyStateMap = [];
70 $legacyStateMapReader = PacketSerializer::decoder(
71 Utils::assumeNotFalse(file_get_contents(Path::join(\pocketmine\BEDROCK_DATA_PATH, "r12_to_current_block_map.bin")), "Missing required resource file"),
72 0,
73 new PacketSerializerContext(GlobalItemTypeDictionary::getInstance()->getDictionary())
74 );
75 $nbtReader = new NetworkNbtSerializer();
76 while(!$legacyStateMapReader->feof()){
77 $id = $legacyStateMapReader->getString();
78 $meta = $legacyStateMapReader->getLShort();
79
80 $offset = $legacyStateMapReader->getOffset();
81 $state = $nbtReader->read($legacyStateMapReader->getBuffer(), $offset)->mustGetCompoundTag();
82 $legacyStateMapReader->setOffset($offset);
83 $legacyStateMap[] = new R12ToCurrentBlockMapEntry($id, $meta, $state);
84 }
85
89 $idToStatesMap = [];
90 foreach($this->bedrockKnownStates as $k => $state){
91 $idToStatesMap[$state->getString("name")][] = $k;
92 }
93 foreach($legacyStateMap as $pair){
94 $id = $legacyIdMap->stringToLegacy($pair->getId());
95 if($id === null){
96 throw new \RuntimeException("No legacy ID matches " . $pair->getId());
97 }
98 $data = $pair->getMeta();
99 if($data > 15){
100 //we can't handle metadata with more than 4 bits
101 continue;
102 }
103 $mappedState = $pair->getBlockState();
104 $mappedName = $mappedState->getString("name");
105 if(!isset($idToStatesMap[$mappedName])){
106 throw new \RuntimeException("Mapped new state does not appear in network table");
107 }
108 foreach($idToStatesMap[$mappedName] as $k){
109 $networkState = $this->bedrockKnownStates[$k];
110 if($mappedState->equals($networkState)){
111 $this->registerMapping($k, $id, $data);
112 continue 2;
113 }
114 }
115 throw new \RuntimeException("Mapped new state does not appear in network table");
116 }
117 }

119 public function toRuntimeId(int $internalStateId) : int{
120 return $this->legacyToRuntimeMap[$internalStateId] ?? $this->legacyToRuntimeMap[BlockLegacyIds::INFO_UPDATE << Block::INTERNAL_METADATA_BITS];
121 }
122
123 public function fromRuntimeId(int $runtimeId) : int{
124 return $this->runtimeToLegacyMap[$runtimeId];
125 }
126
127 private function registerMapping(int $staticRuntimeId, int $legacyId, int $legacyMeta) : void{
128 $this->legacyToRuntimeMap[($legacyId << Block::INTERNAL_METADATA_BITS) | $legacyMeta] = $staticRuntimeId;
129 $this->runtimeToLegacyMap[$staticRuntimeId] = ($legacyId << Block::INTERNAL_METADATA_BITS) | $legacyMeta;
130 }
131
135 public function getBedrockKnownStates() : array{
136 return $this->bedrockKnownStates;
137 }
