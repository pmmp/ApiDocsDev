40 use SingletonTrait;

46 private array $networkIdCache = [];

52 private static function make() : self{
53 $canonicalBlockStatesRaw = Filesystem::fileGetContents(Path::join(\pocketmine\BEDROCK_DATA_PATH, "canonical_block_states.nbt"));
54 $metaMappingRaw = Filesystem::fileGetContents(Path::join(\pocketmine\BEDROCK_DATA_PATH, 'block_state_meta_map.json'));
55 return new self(
56 BlockStateDictionary::loadFromString($canonicalBlockStatesRaw, $metaMappingRaw),
57 GlobalBlockStateHandlers::getSerializer()
58 );
59 }
60
61 public function __construct(
62 private BlockStateDictionary $blockStateDictionary,
63 private BlockStateSerializer $blockStateSerializer
64 ){
65 $this->fallbackStateId = $this->blockStateDictionary->lookupStateIdFromData(
66 new BlockStateData(BlockTypeNames::INFO_UPDATE, [], BlockStateData::CURRENT_VERSION)
67 ) ?? throw new AssumptionFailedError(BlockTypeNames::INFO_UPDATE . " should always exist");
68 //lookup the state data from the dictionary to avoid keeping two copies of the same data around
69 $this->fallbackStateData = $this->blockStateDictionary->getDataFromStateId($this->fallbackStateId) ?? throw new AssumptionFailedError("We just looked up this state data, so it must exist");
70 }
71
72 public function toRuntimeId(int $internalStateId) : int{
73 if(isset($this->networkIdCache[$internalStateId])){
74 return $this->networkIdCache[$internalStateId];
75 }
76
77 try{
78 $blockStateData = $this->blockStateSerializer->serialize($internalStateId);
79
80 $networkId = $this->blockStateDictionary->lookupStateIdFromData($blockStateData);
81 if($networkId === null){
82 throw new AssumptionFailedError("Unmapped blockstate returned by blockstate serializer: " . $blockStateData->toNbt());
83 }
84 }catch(BlockStateSerializeException){
85 //TODO: this will swallow any error caused by invalid block properties; this is not ideal, but it should be
86 //covered by unit tests, so this is probably a safe assumption.
87 $networkId = $this->fallbackStateId;
88 }
89
90 return $this->networkIdCache[$internalStateId] = $networkId;
91 }
92
96 public function toStateData(int $internalStateId) : BlockStateData{
97 //we don't directly use the blockstate serializer here - we can't assume that the network blockstate NBT is the
98 //same as the disk blockstate NBT, in case we decide to have different world version than network version (or in
99 //case someone wants to implement multi version).
100 $networkRuntimeId = $this->toRuntimeId($internalStateId);
101
102 return $this->blockStateDictionary->getDataFromStateId($networkRuntimeId) ?? throw new AssumptionFailedError("We just looked up this state ID, so it must exist");
103 }
104
105 public function getBlockStateDictionary() : BlockStateDictionary{ return $this->blockStateDictionary; }
106
107 public function getFallbackStateData() : BlockStateData{ return $this->fallbackStateData; }
