104 private static function pack(array $items) : array{
106 $result = [];
107
108 foreach($items as $i => $item){
109 foreach($result as $otherItem){
110 if($item->canStackWith($otherItem)){
111 $otherItem->setCount($otherItem->getCount() + $item->getCount());
112 continue 2;
113 }
114 }
115
116 //No matching item found
117 $result[] = clone $item;
118 }
119
120 return $result;
121 }

126 private static function hashOutputs(array $outputs) : string{
127 $outputs = self::pack($outputs);
128 usort($outputs, [self::class, "sort"]);
129 $result = new BinaryStream();
130 foreach($outputs as $o){
131 //count is not written because the outputs might be from multiple repetitions of a single recipe
132 //this reduces the accuracy of the hash, but it won't matter in most cases.
133 $result->putVarInt(morton2d_encode($o->getTypeId(), $o->computeTypeData()));
134 $result->put((new LittleEndianNbtSerializer())->write(new TreeRoot($o->getNamedTag())));
135 }
136
137 return $result->getBuffer();
138 }

257 public function matchBrewingRecipe(Item $input, Item $ingredient) : ?BrewingRecipe{
258 $inputHash = morton2d_encode($input->getTypeId(), $input->computeTypeData());
259 $ingredientHash = morton2d_encode($ingredient->getTypeId(), $ingredient->computeTypeData());
260 $cached = $this->brewingRecipeCache[$inputHash][$ingredientHash] ?? null;
261 if($cached !== null){
262 return $cached;
263 }
264
265 foreach($this->potionContainerChangeRecipes as $recipe){
266 if($recipe->getIngredient()->accepts($ingredient) && $recipe->getResultFor($input) !== null){
267 return $this->brewingRecipeCache[$inputHash][$ingredientHash] = $recipe;
268 }
269 }
270
271 foreach($this->potionTypeRecipes as $recipe){
272 if($recipe->getIngredient()->accepts($ingredient) && $recipe->getResultFor($input) !== null){
273 return $this->brewingRecipeCache[$inputHash][$ingredientHash] = $recipe;
274 }
