180 public static function getScheduledTaskTimings(TaskHandler $task, int $period) : TimingsHandler{
181 self::init();
182 $name = "Task: " . $task->getOwnerName() . " Runnable: " . $task->getTaskName();
183
184 if($period > 0){
185 $name .= "(interval:" . $period . ")";
186 }else{
187 $name .= "(Single)";
188 }
189
190 if(!isset(self::$pluginTaskTimingMap[$name])){
191 self::$pluginTaskTimingMap[$name] = new TimingsHandler($name, self::$schedulerSync);
192 }
193
194 return self::$pluginTaskTimingMap[$name];
195 }
