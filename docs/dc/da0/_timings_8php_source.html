185 public static function getScheduledTaskTimings(TaskHandler $task, int $period) : TimingsHandler{
186 $name = "Task: " . $task->getOwnerName() . " Runnable: " . $task->getTaskName();
187
188 if($period > 0){
189 $name .= "(interval:" . $period . ")";
190 }else{
191 $name .= "(Single)";
192 }
193
194 if(!isset(self::$pluginTaskTimingMap[$name])){
195 self::$pluginTaskTimingMap[$name] = new TimingsHandler($name, self::$schedulerSync);
196 }
197
198 return self::$pluginTaskTimingMap[$name];
199 }

201 public static function getEntityTimings(Entity $entity) : TimingsHandler{
