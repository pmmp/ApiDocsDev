102 public function __construct(Player $player, NetworkSession $session){
103 $this->player = $player;
104 $this->session = $session;
105
106 $this->containerOpenCallbacks = new ObjectSet();
107 $this->containerOpenCallbacks->add(\Closure::fromCallable([self::class, 'createContainerOpen']));
108
109 $this->add(ContainerIds::INVENTORY, $this->player->getInventory());
110 $this->add(ContainerIds::OFFHAND, $this->player->getOffHandInventory());
111 $this->add(ContainerIds::ARMOR, $this->player->getArmorInventory());
112 $this->add(ContainerIds::UI, $this->player->getCursorInventory());
113
114 $this->player->getInventory()->getHeldItemIndexChangeListeners()->add(function() : void{
115 $this->syncSelectedHotbarSlot();
116 });

139 public function onTransactionStart(InventoryTransaction $tx) : void{
140 foreach($tx->getActions() as $action){
141 if($action instanceof SlotChangeAction && ($windowId = $this->getWindowId($action->getInventory())) !== null){
142 //in some cases the inventory might not have a window ID, but still be referenced by a transaction (e.g. crafting grid changes), so we can't unconditionally record the change here or we might leak things
143 $this->initiatedSlotChanges[$windowId][$action->getSlot()] = $action->getTargetItem();
144 }
145 }

148 public function onCurrentWindowChange(Inventory $inventory) : void{
149 $this->onCurrentWindowRemove();
150 $this->add($this->lastInventoryNetworkId = max(ContainerIds::FIRST, ($this->lastInventoryNetworkId + 1) % self::RESERVED_WINDOW_ID_RANGE_START), $inventory);
151
152 foreach($this->containerOpenCallbacks as $callback){
153 $pks = $callback($this->lastInventoryNetworkId, $inventory);
154 if($pks !== null){
155 foreach($pks as $pk){
156 $this->session->sendDataPacket($pk);
157 }
158 $this->syncContents($inventory);
159 return;
160 }
161 }
162 throw new \LogicException("Unsupported inventory type");
163 }

172 protected static function createContainerOpen(int $id, Inventory $inv) : ?array{
173 //TODO: we should be using some kind of tagging system to identify the types. Instanceof is flaky especially
174 //if the class isn't final, not to mention being inflexible.
175 if($inv instanceof BlockInventory){
176 $blockPosition = BlockPosition::fromVector3($inv->getHolder());
177 $windowType = match(true){
178 $inv instanceof LoomInventory => WindowTypes::LOOM,
179 $inv instanceof FurnaceInventory => match($inv->getFurnaceType()->id()){
180 FurnaceType::FURNACE()->id() => WindowTypes::FURNACE,
181 FurnaceType::BLAST_FURNACE()->id() => WindowTypes::BLAST_FURNACE,
182 FurnaceType::SMOKER()->id() => WindowTypes::SMOKER,
183 default => throw new AssumptionFailedError("Unreachable")
184 },
185 $inv instanceof EnchantInventory => WindowTypes::ENCHANTMENT,
186 $inv instanceof BrewingStandInventory => WindowTypes::BREWING_STAND,
187 $inv instanceof AnvilInventory => WindowTypes::ANVIL,
188 $inv instanceof HopperInventory => WindowTypes::HOPPER,
189 $inv instanceof CraftingTableInventory => WindowTypes::WORKBENCH,
190 default => WindowTypes::CONTAINER
191 };
192 return [ContainerOpenPacket::blockInv($id, $windowType, $blockPosition)];
193 }
194 return null;
195 }

197 public function onClientOpenMainInventory() : void{
198 $id = self::HARDCODED_INVENTORY_WINDOW_ID;
199 if(!isset($this->openHardcodedWindows[$id])){
200 //TODO: HACK! this restores 1.14ish behaviour, but this should be able to be listened to and
201 //controlled by plugins. However, the player is always a subscriber to their own inventory so it
202 //doesn't integrate well with the regular container system right now.
203 $this->openHardcodedWindows[$id] = true;
204 $this->session->sendDataPacket(ContainerOpenPacket::entityInv(
205 InventoryManager::HARDCODED_INVENTORY_WINDOW_ID,
206 WindowTypes::INVENTORY,
207 $this->player->getId()
208 ));
209 }

212 public function onCurrentWindowRemove() : void{
213 if(isset($this->windowMap[$this->lastInventoryNetworkId])){
214 $this->remove($this->lastInventoryNetworkId);
215 $this->session->sendDataPacket(ContainerClosePacket::create($this->lastInventoryNetworkId, true));
216 }

219 public function onClientRemoveWindow(int $id) : void{
220 if(isset($this->openHardcodedWindows[$id])){
221 unset($this->openHardcodedWindows[$id]);
222 }elseif($id === $this->lastInventoryNetworkId){
223 $this->remove($id);
224 $this->player->removeCurrentWindow();
225 }else{
226 $this->session->getLogger()->debug("Attempted to close inventory with network ID $id, but current is $this->lastInventoryNetworkId");
227 }
228
229 //Always send this, even if no window matches. If we told the client to close a window, it will behave as if it
230 //initiated the close and expect an ack.
231 $this->session->sendDataPacket(ContainerClosePacket::create($id, false));
232 }

234 public function syncSlot(Inventory $inventory, int $slot) : void{
235 $windowId = $this->getWindowId($inventory);
236 if($windowId !== null){
237 $currentItem = $inventory->getItem($slot);
238 $clientSideItem = $this->initiatedSlotChanges[$windowId][$slot] ?? null;
239 if($clientSideItem === null || !$clientSideItem->equalsExact($currentItem)){
240 $itemStackWrapper = ItemStackWrapper::legacy(TypeConverter::getInstance()->coreItemStackToNet($currentItem));
241 if($windowId === ContainerIds::OFFHAND){
242 //TODO: HACK!
243 //The client may sometimes ignore the InventorySlotPacket for the offhand slot.
244 //This can cause a lot of problems (totems, arrows, and more...).
245 //The workaround is to send an InventoryContentPacket instead
246 //BDS (Bedrock Dedicated Server) also seems to work this way.
247 $this->session->sendDataPacket(InventoryContentPacket::create($windowId, [$itemStackWrapper]));
248 }else{
249 $this->session->sendDataPacket(InventorySlotPacket::create($windowId, $slot, $itemStackWrapper));
250 }
251 }
252 unset($this->initiatedSlotChanges[$windowId][$slot]);
253 }
254 }

256 public function syncContents(Inventory $inventory) : void{
257 $windowId = $this->getWindowId($inventory);
258 if($windowId !== null){
259 unset($this->initiatedSlotChanges[$windowId]);
260 $typeConverter = TypeConverter::getInstance();
261 if($windowId === ContainerIds::UI){
262 //TODO: HACK!
263 //Since 1.13, cursor is now part of a larger "UI inventory", and sending contents for this larger inventory does
264 //not work the way it's intended to. Even if it did, it would be necessary to send all 51 slots just to update
265 //this one, which is just not worth it.
266 //This workaround isn't great, but it's at least simple.
267 $this->session->sendDataPacket(InventorySlotPacket::create(
268 $windowId,
269 0,
270 ItemStackWrapper::legacy($typeConverter->coreItemStackToNet($inventory->getItem(0)))
271 ));
272 }else{
273 $this->session->sendDataPacket(InventoryContentPacket::create($windowId, array_map(function(Item $itemStack) use ($typeConverter) : ItemStackWrapper{
274 return ItemStackWrapper::legacy($typeConverter->coreItemStackToNet($itemStack));
275 }, $inventory->getContents(true))));
276 }
277 }
278 }

297 public function syncSelectedHotbarSlot() : void{
298 $selected = $this->player->getInventory()->getHeldItemIndex();
299 if($selected !== $this->clientSelectedHotbarSlot){
300 $this->session->sendDataPacket(MobEquipmentPacket::create(
301 $this->player->getId(),
302 ItemStackWrapper::legacy(TypeConverter::getInstance()->coreItemStackToNet($this->player->getInventory()->getItemInHand())),
303 $selected,
304 $selected,
305 ContainerIds::INVENTORY
306 ));
307 $this->clientSelectedHotbarSlot = $selected;
308 }
309 }

317 }, $this->player->isSpectator() ? [] : CreativeInventory::getInstance()->getAll())));
