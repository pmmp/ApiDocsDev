137 $entry = new ComplexWindowMapEntry($inventory, is_int($slotMap) ? [$slotMap => 0] : $slotMap);

145 $inventory = $this->windowMap[$id];
146 $splObjectId = spl_object_id($inventory);
147 unset($this->windowMap[$id], $this->initiatedSlotChanges[$id], $this->complexWindows[$splObjectId]);
148 foreach($this->complexSlotToWindowMap as $netSlot => $entry){

166 public function locateWindowAndSlot(int $windowId, int $netSlotId) : ?array{
167 if($windowId === ContainerIds::UI){
168 $entry = $this->complexSlotToWindowMap[$netSlotId] ?? null;
169 if($entry === null){
170 return null;
171 }
172 $coreSlotId = $entry->mapNetToCore($netSlotId);
173 return $coreSlotId !== null ? [$entry->getInventory(), $coreSlotId] : null;
174 }
175 if(isset($this->windowMap[$windowId])){
176 return [$this->windowMap[$windowId], $netSlotId];
177 }
178 return null;
179 }

181 public function onTransactionStart(InventoryTransaction $tx) : void{
182 foreach($tx->getActions() as $action){
183 if($action instanceof SlotChangeAction && ($windowId = $this->getWindowId($action->getInventory())) !== null){
184 //in some cases the inventory might not have a window ID, but still be referenced by a transaction (e.g. crafting grid changes), so we can't unconditionally record the change here or we might leak things
185 $this->initiatedSlotChanges[$windowId][$action->getSlot()] = $action->getTargetItem();
186 }

194 public function addPredictedSlotChanges(array $networkInventoryActions) : void{
195 foreach($networkInventoryActions as $action){
196 if($action->sourceType === NetworkInventoryAction::SOURCE_CONTAINER && isset($this->windowMap[$action->windowId])){
197 //this won't cover stuff like crafting grid due to too much magic
198 try{
199 $item = TypeConverter::getInstance()->netItemStackToCore($action->newItem->getItemStack());
200 }catch(TypeConversionException $e){
201 throw new PacketHandlingException($e->getMessage(), 0, $e);
202 }
203 $this->initiatedSlotChanges[$action->windowId][$action->inventorySlot] = $item;
204 }

237 return match(true){

249 public function onCurrentWindowChange(Inventory $inventory) : void{
250 $this->onCurrentWindowRemove();
251
252 $this->openWindowDeferred(function() use ($inventory) : void{
253 $windowId = $this->addDynamic($inventory);
254 if(($slotMap = $this->createComplexSlotMapping($inventory)) !== null){
255 $this->addComplex($slotMap, $inventory);
256 }
257
258 foreach($this->containerOpenCallbacks as $callback){
259 $pks = $callback($windowId, $inventory);
260 if($pks !== null){
261 foreach($pks as $pk){
262 $this->session->sendDataPacket($pk);
263 }
264 $this->syncContents($inventory);
265 return;
266 }
267 }
268 throw new \LogicException("Unsupported inventory type");
269 });
270 }

279 protected static function createContainerOpen(int $id, Inventory $inv) : ?array{
280 //TODO: we should be using some kind of tagging system to identify the types. Instanceof is flaky especially
281 //if the class isn't final, not to mention being inflexible.
282 if($inv instanceof BlockInventory){
283 $blockPosition = BlockPosition::fromVector3($inv->getHolder());
284 $windowType = match(true){
285 $inv instanceof LoomInventory => WindowTypes::LOOM,
286 $inv instanceof FurnaceInventory => match($inv->getFurnaceType()->id()){
287 FurnaceType::FURNACE()->id() => WindowTypes::FURNACE,
288 FurnaceType::BLAST_FURNACE()->id() => WindowTypes::BLAST_FURNACE,
289 FurnaceType::SMOKER()->id() => WindowTypes::SMOKER,
290 default => throw new AssumptionFailedError("Unreachable")
291 },
292 $inv instanceof EnchantInventory => WindowTypes::ENCHANTMENT,
293 $inv instanceof BrewingStandInventory => WindowTypes::BREWING_STAND,
294 $inv instanceof AnvilInventory => WindowTypes::ANVIL,
295 $inv instanceof HopperInventory => WindowTypes::HOPPER,
296 $inv instanceof CraftingTableInventory => WindowTypes::WORKBENCH,
297 $inv instanceof StonecutterInventory => WindowTypes::STONECUTTER,
298 $inv instanceof CartographyTableInventory => WindowTypes::CARTOGRAPHY,
299 $inv instanceof SmithingTableInventory => WindowTypes::SMITHING_TABLE,
300 default => WindowTypes::CONTAINER
301 };
302 return [ContainerOpenPacket::blockInv($id, $windowType, $blockPosition)];
303 }
304 return null;
305 }

321 public function onCurrentWindowRemove() : void{
322 if(isset($this->windowMap[$this->lastInventoryNetworkId])){
323 $this->remove($this->lastInventoryNetworkId);
324 $this->session->sendDataPacket(ContainerClosePacket::create($this->lastInventoryNetworkId, true));
325 if($this->pendingCloseWindowId !== null){
326 throw new AssumptionFailedError("We should not have opened a new window while a window was waiting to be closed");
327 }
328 $this->pendingCloseWindowId = $this->lastInventoryNetworkId;
329 }

332 public function onClientRemoveWindow(int $id) : void{
333 if($id === $this->lastInventoryNetworkId){
334 if(isset($this->windowMap[$id]) && $id !== $this->pendingCloseWindowId){
335 $this->remove($id);
336 $this->player->removeCurrentWindow();
337 }
338 }else{
339 $this->session->getLogger()->debug("Attempted to close inventory with network ID $id, but current is $this->lastInventoryNetworkId");
340 }
341
342 //Always send this, even if no window matches. If we told the client to close a window, it will behave as if it
343 //initiated the close and expect an ack.
344 $this->session->sendDataPacket(ContainerClosePacket::create($id, false));
345
346 if($this->pendingCloseWindowId === $id){
347 $this->pendingCloseWindowId = null;
348 if($this->pendingOpenWindowCallback !== null){
349 $this->session->getLogger()->debug("Opening deferred window after close ack of window $id");
350 ($this->pendingOpenWindowCallback)();
351 $this->pendingOpenWindowCallback = null;
352 }
353 }

356 public function syncSlot(Inventory $inventory, int $slot) : void{
357 $slotMap = $this->complexWindows[spl_object_id($inventory)] ?? null;
358 if($slotMap !== null){
359 $windowId = ContainerIds::UI;
360 $netSlot = $slotMap->mapCoreToNet($slot) ?? null;
361 }else{
362 $windowId = $this->getWindowId($inventory);
363 $netSlot = $slot;
364 }
365 if($windowId !== null && $netSlot !== null){
366 $currentItem = $inventory->getItem($slot);
367 $clientSideItem = $this->initiatedSlotChanges[$windowId][$netSlot] ?? null;
368 if($clientSideItem === null || !$clientSideItem->equalsExact($currentItem)){
369 $itemStackWrapper = ItemStackWrapper::legacy(TypeConverter::getInstance()->coreItemStackToNet($currentItem));
370 if($windowId === ContainerIds::OFFHAND){
371 //TODO: HACK!
372 //The client may sometimes ignore the InventorySlotPacket for the offhand slot.
373 //This can cause a lot of problems (totems, arrows, and more...).
374 //The workaround is to send an InventoryContentPacket instead
375 //BDS (Bedrock Dedicated Server) also seems to work this way.
376 $this->session->sendDataPacket(InventoryContentPacket::create($windowId, [$itemStackWrapper]));
377 }else{
378 $this->session->sendDataPacket(InventorySlotPacket::create($windowId, $netSlot, $itemStackWrapper));
379 }
380 }
381 unset($this->initiatedSlotChanges[$windowId][$netSlot]);
382 }
383 }

385 public function syncContents(Inventory $inventory) : void{
386 $slotMap = $this->complexWindows[spl_object_id($inventory)] ?? null;
387 if($slotMap !== null){
388 $windowId = ContainerIds::UI;
389 }else{
390 $windowId = $this->getWindowId($inventory);
391 }
392 $typeConverter = TypeConverter::getInstance();
393 if($windowId !== null){
394 if($slotMap !== null){
395 foreach($inventory->getContents(true) as $slotId => $item){
396 $packetSlot = $slotMap->mapCoreToNet($slotId) ?? null;
397 if($packetSlot === null){
398 continue;
399 }
400 unset($this->initiatedSlotChanges[$windowId][$packetSlot]);
401 $this->session->sendDataPacket(InventorySlotPacket::create(
402 $windowId,
403 $packetSlot,
404 ItemStackWrapper::legacy($typeConverter->coreItemStackToNet($inventory->getItem($slotId)))
405 ));
406 }
407 }else{
408 unset($this->initiatedSlotChanges[$windowId]);
409 $this->session->sendDataPacket(InventoryContentPacket::create($windowId, array_map(function(Item $itemStack) use ($typeConverter) : ItemStackWrapper{
410 return ItemStackWrapper::legacy($typeConverter->coreItemStackToNet($itemStack));
411 }, $inventory->getContents(true))));
412 }
413 }

425 public function syncMismatchedPredictedSlotChanges() : void{
426 foreach($this->initiatedSlotChanges as $windowId => $slots){
427 foreach($slots as $netSlot => $expectedItem){
428 $located = $this->locateWindowAndSlot($windowId, $netSlot);
429 if($located === null){
430 continue;
431 }
432 [$inventory, $slot] = $located;
433
434 if(!$inventory->slotExists($slot)){
435 continue; //TODO: size desync ???
436 }
437 $actualItem = $inventory->getItem($slot);
438 if(!$actualItem->equalsExact($expectedItem)){
439 $this->session->getLogger()->debug("Detected prediction mismatch in inventory " . get_class($inventory) . "#" . spl_object_id($inventory) . " slot $slot");
440 $this->syncSlot($inventory, $slot);
441 }
442 }
443 }
444
445 $this->initiatedSlotChanges = [];
446 }
