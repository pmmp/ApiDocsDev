69 private static function make() : self{
70 $data = Utils::assumeNotFalse(file_get_contents(Path::join(\pocketmine\BEDROCK_DATA_PATH, 'r16_to_current_item_map.json')), "Missing required resource file");
71 $json = json_decode($data, true);
72 if(!is_array($json) || !isset($json["simple"], $json["complex"]) || !is_array($json["simple"]) || !is_array($json["complex"])){
73 throw new AssumptionFailedError("Invalid item table format");
74 }
75
76 $legacyStringToIntMap = LegacyItemIdToStringIdMap::getInstance();
77
79 $simpleMappings = [];
80 foreach($json["simple"] as $oldId => $newId){
81 if(!is_string($oldId) || !is_string($newId)){
82 throw new AssumptionFailedError("Invalid item table format");
83 }
84 $intId = $legacyStringToIntMap->stringToLegacy($oldId);
85 if($intId === null){
86 //new item without a fixed legacy ID - we can't handle this right now
87 continue;
88 }
89 $simpleMappings[$newId] = $intId;
90 }
91 foreach(Utils::stringifyKeys($legacyStringToIntMap->getStringToLegacyMap()) as $stringId => $intId){
92 if(isset($simpleMappings[$stringId])){
93 throw new \UnexpectedValueException("Old ID $stringId collides with new ID");
94 }
95 $simpleMappings[$stringId] = $intId;
96 }
97
99 $complexMappings = [];
100 foreach($json["complex"] as $oldId => $map){
101 if(!is_string($oldId) || !is_array($map)){
102 throw new AssumptionFailedError("Invalid item table format");
103 }
104 foreach($map as $meta => $newId){
105 if(!is_numeric($meta) || !is_string($newId)){
106 throw new AssumptionFailedError("Invalid item table format");
107 }
108 $intId = $legacyStringToIntMap->stringToLegacy($oldId);
109 if($intId === null){
110 //new item without a fixed legacy ID - we can't handle this right now
111 continue;
112 }
113 $complexMappings[$newId] = [$intId, (int) $meta];
114 }
115 }
116
117 return new self(GlobalItemTypeDictionary::getInstance()->getDictionary(), $simpleMappings, $complexMappings);
118 }

126 public function __construct(ItemTypeDictionary $dictionary, array $simpleMappings, array $complexMappings){
127 foreach($dictionary->getEntries() as $entry){
128 $stringId = $entry->getStringId();
129 $netId = $entry->getNumericId();
130 if(isset($complexMappings[$stringId])){
131 [$id, $meta] = $complexMappings[$stringId];
132 $this->complexCoreToNetMapping[$id][$meta] = $netId;
133 $this->complexNetToCoreMapping[$netId] = [$id, $meta];
134 }elseif(isset($simpleMappings[$stringId])){
135 $this->simpleCoreToNetMapping[$simpleMappings[$stringId]] = $netId;
136 $this->simpleNetToCoreMapping[$netId] = $simpleMappings[$stringId];
137 }else{
138 //not all items have a legacy mapping - for now, we only support the ones that do
139 continue;
140 }
141 }
142 }

148 public function toNetworkIdQuiet(int $internalId, int $internalMeta) : ?array{
149 if($internalMeta === -1){
150 $internalMeta = 0x7fff;
151 }
152 if(isset($this->complexCoreToNetMapping[$internalId][$internalMeta])){
153 return [$this->complexCoreToNetMapping[$internalId][$internalMeta], 0];
154 }
155 if(array_key_exists($internalId, $this->simpleCoreToNetMapping)){
156 return [$this->simpleCoreToNetMapping[$internalId], $internalMeta];
157 }
158
159 return null;
160 }

166 public function toNetworkId(int $internalId, int $internalMeta) : array{
167 return $this->toNetworkIdQuiet($internalId, $internalMeta) ??
168 throw new \InvalidArgumentException("Unmapped ID/metadata combination $internalId:$internalMeta");
169 }
170
176 public function fromNetworkId(int $networkId, int $networkMeta, ?bool &$isComplexMapping = null) : array{
177 if(isset($this->complexNetToCoreMapping[$networkId])){
178 if($networkMeta !== 0){
179 throw new TypeConversionException("Unexpected non-zero network meta on complex item mapping");
180 }
181 $isComplexMapping = true;
182 return $this->complexNetToCoreMapping[$networkId];
183 }
184 $isComplexMapping = false;
185 if(isset($this->simpleNetToCoreMapping[$networkId])){
186 return [$this->simpleNetToCoreMapping[$networkId], $networkMeta];
187 }
188 throw new TypeConversionException("Unmapped network ID/metadata combination $networkId:$networkMeta");
189 }
190
196 public function fromNetworkIdWithWildcardHandling(int $networkId, int $networkMeta) : array{
197 $isComplexMapping = false;
198 if($networkMeta !== 0x7fff){
199 return $this->fromNetworkId($networkId, $networkMeta);
200 }
201 [$id, $meta] = $this->fromNetworkId($networkId, 0, $isComplexMapping);
202 return [$id, $isComplexMapping ? $meta : -1];
203 }
