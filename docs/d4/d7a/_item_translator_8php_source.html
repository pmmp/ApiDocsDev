80 public function toNetworkId(Item $item) : array{
81 //TODO: we should probably come up with a cache for this
82
83 $itemData = $this->itemSerializer->serializeType($item);
84
85 $numericId = $this->itemTypeDictionary->fromStringId($itemData->getName());
86 $blockStateData = $itemData->getBlock();
87
88 if($blockStateData !== null){
89 $blockRuntimeId = $this->blockStateDictionary->lookupStateIdFromData($blockStateData);
90 if($blockRuntimeId === null){
91 throw new AssumptionFailedError("Unmapped blockstate returned by blockstate serializer: " . $blockStateData->toNbt());
92 }
93 }else{
94 $blockRuntimeId = self::NO_BLOCK_RUNTIME_ID; //this is technically a valid block runtime ID, but is used to represent "no block" (derp mojang)
95 }
96
97 return [$numericId, $itemData->getMeta(), $blockRuntimeId];
98 }

103 public function toNetworkNbt(Item $item) : CompoundTag{
104 //TODO: this relies on the assumption that network item NBT is the same as disk item NBT, which may not always
105 //be true - if we stick on an older world version while updating network version, this could be a problem (and
106 //may be a problem for multi version implementations)
107 return $this->itemSerializer->serializeStack($item)->toNbt();
108 }
109
113 public function fromNetworkId(int $networkId, int $networkMeta, int $networkBlockRuntimeId) : Item{
114 try{
115 $stringId = $this->itemTypeDictionary->fromIntId($networkId);
116 }catch(\InvalidArgumentException $e){
117 //TODO: a quiet version of fromIntId() would be better than catching InvalidArgumentException
118 throw TypeConversionException::wrap($e, "Invalid network itemstack ID $networkId");
119 }
120
121 $blockStateData = null;
122 if($networkBlockRuntimeId !== self::NO_BLOCK_RUNTIME_ID){
123 $blockStateData = $this->blockStateDictionary->getDataFromStateId($networkBlockRuntimeId);
124 if($blockStateData === null){
125 throw new TypeConversionException("Blockstate runtimeID $networkBlockRuntimeId does not correspond to any known blockstate");
126 }
127 }
128
129 try{
130 return $this->itemDeserializer->deserializeType(new SavedItemData($stringId, $networkMeta, $blockStateData));
131 }catch(ItemTypeDeserializeException $e){
132 throw TypeConversionException::wrap($e, "Invalid network itemstack data");
133 }
134 }
