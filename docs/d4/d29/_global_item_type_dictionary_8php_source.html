42 private static function make() : self{
43 $data = Utils::assumeNotFalse(file_get_contents(Path::join(\pocketmine\BEDROCK_DATA_PATH, 'required_item_list.json')), "Missing required resource file");
44 $table = json_decode($data, true);
45 if(!is_array($table)){
46 throw new AssumptionFailedError("Invalid item list format");
47 }
48
49 $params = [];
50 foreach($table as $name => $entry){
51 if(!is_array($entry) || !is_string($name) || !isset($entry["component_based"], $entry["runtime_id"]) || !is_bool($entry["component_based"]) || !is_int($entry["runtime_id"])){
52 throw new AssumptionFailedError("Invalid item list format");
53 }
54 $params[] = new ItemTypeEntry($name, $entry["runtime_id"], $entry["component_based"]);
55 }
56 return new self(new ItemTypeDictionary($params));
57 }
