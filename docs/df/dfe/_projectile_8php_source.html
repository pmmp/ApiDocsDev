169 protected function move(float $dx, float $dy, float $dz) : void{
170 $this->blocksAround = null;
171
172 Timings::$entityMove->startTiming();
173
174 $start = $this->location->asVector3();
175 $end = $start->add($dx, $dy, $dz);
176
177 $blockHit = null;
178 $entityHit = null;
179 $hitResult = null;
180
181 foreach(VoxelRayTrace::betweenPoints($start, $end) as $vector3){
182 $block = $this->getWorld()->getBlockAt($vector3->x, $vector3->y, $vector3->z);
183
184 $blockHitResult = $this->calculateInterceptWithBlock($block, $start, $end);
185 if($blockHitResult !== null){
186 $end = $blockHitResult->hitVector;
187 $blockHit = $block;
188 $hitResult = $blockHitResult;
189 break;
190 }
191 }
192
193 $entityDistance = PHP_INT_MAX;
194
195 $newDiff = $end->subtractVector($start);
196 foreach($this->getWorld()->getCollidingEntities($this->boundingBox->addCoord($newDiff->x, $newDiff->y, $newDiff->z)->expand(1, 1, 1), $this) as $entity){
197 if($entity->getId() === $this->getOwningEntityId() && $this->ticksLived < 5){
198 continue;
199 }
200
201 $entityBB = $entity->boundingBox->expandedCopy(0.3, 0.3, 0.3);
202 $entityHitResult = $entityBB->calculateIntercept($start, $end);
203
204 if($entityHitResult === null){
205 continue;
206 }
207
208 $distance = $this->location->distanceSquared($entityHitResult->hitVector);
209
210 if($distance < $entityDistance){
211 $entityDistance = $distance;
212 $entityHit = $entity;
213 $hitResult = $entityHitResult;
214 $end = $entityHitResult->hitVector;
215 }
216 }
217
218 $this->location = Location::fromObject(
219 $end,
220 $this->location->world,
221 $this->location->yaw,
222 $this->location->pitch
223 );
224 $this->recalculateBoundingBox();
225
226 if($hitResult !== null){
228 $ev = null;
229 if($entityHit !== null){
230 $ev = new ProjectileHitEntityEvent($this, $hitResult, $entityHit);
231 }elseif($blockHit !== null){
232 $ev = new ProjectileHitBlockEvent($this, $hitResult, $blockHit);
233 }else{
234 assert(false, "unknown hit type");
235 }
236
237 if($ev !== null){
238 $ev->call();
239 $this->onHit($ev);
240
241 if($ev instanceof ProjectileHitEntityEvent){
242 $this->onHitEntity($ev->getEntityHit(), $ev->getRayTraceResult());
243 }elseif($ev instanceof ProjectileHitBlockEvent){
244 $this->onHitBlock($ev->getBlockHit(), $ev->getRayTraceResult());
245 }
246 }
247
248 $this->isCollided = $this->onGround = true;
249 $this->motion = new Vector3(0, 0, 0);
250 }else{
251 $this->isCollided = $this->onGround = false;
252 $this->blockHit = null;
253
254 //recompute angles...
255 $f = sqrt(($this->motion->x ** 2) + ($this->motion->z ** 2));
256 $this->setRotation(
257 atan2($this->motion->x, $this->motion->z) * 180 / M_PI,
258 atan2($this->motion->y, $f) * 180 / M_PI
259 );
260 }
261
262 $this->getWorld()->onEntityMoved($this);
263 $this->checkBlockIntersections();
264
265 Timings::$entityMove->stopTiming();
266 }

290 protected function onHitEntity(Entity $entityHit, RayTraceResult $hitResult) : void{
291 $damage = $this->getResultDamage();
292
293 if($damage >= 0){
294 if($this->getOwningEntity() === null){
295 $ev = new EntityDamageByEntityEvent($this, $entityHit, EntityDamageEvent::CAUSE_PROJECTILE, $damage);
296 }else{
297 $ev = new EntityDamageByChildEntityEvent($this->getOwningEntity(), $this, $entityHit, EntityDamageEvent::CAUSE_PROJECTILE, $damage);
298 }
299
300 $entityHit->attack($ev);
301
302 if($this->isOnFire()){
303 $ev = new EntityCombustByEntityEvent($this, $entityHit, 5);
304 $ev->call();
305 if(!$ev->isCancelled()){
306 $entityHit->setOnFire($ev->getDuration());
307 }
308 }
309 }
310
311 $this->flagForDespawn();
312 }
