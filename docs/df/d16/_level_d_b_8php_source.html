164 protected function deserializePaletted(BinaryStream $stream) : PalettedBlockArray{
165 $bitsPerBlock = $stream->getByte() >> 1;
166
167 try{
168 $words = $stream->get(PalettedBlockArray::getExpectedWordArraySize($bitsPerBlock));
169 }catch(\InvalidArgumentException $e){
170 throw new CorruptedChunkException("Failed to deserialize paletted storage: " . $e->getMessage(), 0, $e);
171 }
172 $nbt = new LittleEndianNbtSerializer();
173 $palette = [];
174 $idMap = LegacyBlockIdToStringIdMap::getInstance();
175 for($i = 0, $paletteSize = $stream->getLInt(); $i < $paletteSize; ++$i){
176 try{
177 $offset = $stream->getOffset();
178
179 $tag = $nbt->read($stream->getBuffer(), $offset)->mustGetCompoundTag();
180 $stream->setOffset($offset);
181
182 $id = $idMap->stringToLegacy($tag->getString("name")) ?? BlockLegacyIds::INFO_UPDATE;
183 $data = $tag->getShort("val");
184 $palette[] = ($id << Block::INTERNAL_METADATA_BITS) | $data;
185 }catch(NbtException $e){
186 throw new CorruptedChunkException("Invalid blockstate NBT at offset $i in paletted storage: " . $e->getMessage(), 0, $e);
187 }
188 }
189
190 //TODO: exceptions
191 return PalettedBlockArray::fromData($bitsPerBlock, $words, $palette);
192 }

194 protected static function deserializeExtraDataKey(int $chunkVersion, int $key, ?int &$x, ?int &$y, ?int &$z) : void{
195 if($chunkVersion >= ChunkVersion::v1_0_0){
196 $x = ($key >> 12) & 0xf;
197 $z = ($key >> 8) & 0xf;
198 $y = $key & 0xff;
199 }else{ //pre-1.0, 7 bits were used because the build height limit was lower
200 $x = ($key >> 11) & 0xf;
201 $z = ($key >> 7) & 0xf;
202 $y = $key & 0x7f;
203 }
204 }

209 protected function deserializeLegacyExtraData(string $index, int $chunkVersion) : array{
210 if(($extraRawData = $this->db->get($index . ChunkDataKey::LEGACY_BLOCK_EXTRA_DATA)) === false || $extraRawData === ""){
211 return [];
212 }
213
215 $extraDataLayers = [];
216 $binaryStream = new BinaryStream($extraRawData);
217 $count = $binaryStream->getLInt();
218 for($i = 0; $i < $count; ++$i){
219 $key = $binaryStream->getLInt();
220 $value = $binaryStream->getLShort();
221
222 self::deserializeExtraDataKey($chunkVersion, $key, $x, $fullY, $z);
223
224 $ySub = ($fullY >> SubChunk::COORD_BIT_SIZE);
225 $y = $key & SubChunk::COORD_MASK;
226
227 $blockId = $value & 0xff;
228 $blockData = ($value >> 8) & 0xf;
229 if(!isset($extraDataLayers[$ySub])){
230 $extraDataLayers[$ySub] = new PalettedBlockArray(BlockLegacyIds::AIR << Block::INTERNAL_METADATA_BITS);
231 }
232 $extraDataLayers[$ySub]->set($x, $y, $z, ($blockId << Block::INTERNAL_METADATA_BITS) | $blockData);
233 }
234
235 return $extraDataLayers;
236 }

241 public function loadChunk(int $chunkX, int $chunkZ) : ?ChunkData{
242 $index = LevelDB::chunkIndex($chunkX, $chunkZ);
243
244 $chunkVersionRaw = $this->db->get($index . ChunkDataKey::OLD_VERSION);
245 if($chunkVersionRaw === false){
246 return null;
247 }
248
250 $subChunks = [];
251
253 $biomeArray = null;
254
255 $chunkVersion = ord($chunkVersionRaw);
256 $hasBeenUpgraded = $chunkVersion < self::CURRENT_LEVEL_CHUNK_VERSION;
257
258 switch($chunkVersion){
259 case ChunkVersion::v1_12_0_4_beta:
260 case ChunkVersion::v1_11_1:
261 case ChunkVersion::v1_11_0_4_beta:
262 case ChunkVersion::v1_11_0_3_beta:
263 case ChunkVersion::v1_11_0_1_beta:
264 case ChunkVersion::v1_9_0:
265 case ChunkVersion::v1_8_0:
266 case ChunkVersion::v1_2_13:
267 case ChunkVersion::v1_2_0:
268 case ChunkVersion::v1_2_0_2_beta:
269 case ChunkVersion::v1_1_0:
270 //TODO: check beds
271 case ChunkVersion::v1_0_0:
272 $convertedLegacyExtraData = $this->deserializeLegacyExtraData($index, $chunkVersion);
273
274 for($y = Chunk::MIN_SUBCHUNK_INDEX; $y <= Chunk::MAX_SUBCHUNK_INDEX; ++$y){
275 if(($data = $this->db->get($index . ChunkDataKey::SUBCHUNK . chr($y))) === false){
276 continue;
277 }
278
279 $binaryStream = new BinaryStream($data);
280 if($binaryStream->feof()){
281 throw new CorruptedChunkException("Unexpected empty data for subchunk $y");
282 }
283 $subChunkVersion = $binaryStream->getByte();
284 if($subChunkVersion < self::CURRENT_LEVEL_SUBCHUNK_VERSION){
285 $hasBeenUpgraded = true;
286 }
287
288 switch($subChunkVersion){
289 case SubChunkVersion::CLASSIC:
290 case SubChunkVersion::CLASSIC_BUG_2: //these are all identical to version 0, but vanilla respects these so we should also
291 case SubChunkVersion::CLASSIC_BUG_3:
292 case SubChunkVersion::CLASSIC_BUG_4:
293 case SubChunkVersion::CLASSIC_BUG_5:
294 case SubChunkVersion::CLASSIC_BUG_6:
295 case SubChunkVersion::CLASSIC_BUG_7:
296 try{
297 $blocks = $binaryStream->get(4096);
298 $blockData = $binaryStream->get(2048);
299
300 if($chunkVersion < ChunkVersion::v1_1_0){
301 $binaryStream->get(4096); //legacy light info, discard it
302 $hasBeenUpgraded = true;
303 }
304 }catch(BinaryDataException $e){
305 throw new CorruptedChunkException($e->getMessage(), 0, $e);
306 }
307
308 $storages = [SubChunkConverter::convertSubChunkXZY($blocks, $blockData)];
309 if(isset($convertedLegacyExtraData[$y])){
310 $storages[] = $convertedLegacyExtraData[$y];
311 }
312
313 $subChunks[$y] = new SubChunk(BlockLegacyIds::AIR << Block::INTERNAL_METADATA_BITS, $storages);
314 break;
315 case SubChunkVersion::PALETTED_SINGLE:
316 $storages = [$this->deserializePaletted($binaryStream)];
317 if(isset($convertedLegacyExtraData[$y])){
318 $storages[] = $convertedLegacyExtraData[$y];
319 }
320 $subChunks[$y] = new SubChunk(BlockLegacyIds::AIR << Block::INTERNAL_METADATA_BITS, $storages);
321 break;
322 case SubChunkVersion::PALETTED_MULTI:
323 //legacy extradata layers intentionally ignored because they aren't supposed to exist in v8
324 $storageCount = $binaryStream->getByte();
325 if($storageCount > 0){
326 $storages = [];
327
328 for($k = 0; $k < $storageCount; ++$k){
329 $storages[] = $this->deserializePaletted($binaryStream);
330 }
331 $subChunks[$y] = new SubChunk(BlockLegacyIds::AIR << Block::INTERNAL_METADATA_BITS, $storages);
332 }
333 break;
334 default:
335 //TODO: set chunks read-only so the version on disk doesn't get overwritten
336 throw new CorruptedChunkException("don't know how to decode LevelDB subchunk format version $subChunkVersion");
337 }
338 }
339
340 if(($maps2d = $this->db->get($index . ChunkDataKey::HEIGHTMAP_AND_2D_BIOMES)) !== false){
341 $binaryStream = new BinaryStream($maps2d);
342
343 try{
344 $binaryStream->get(512); //heightmap, discard it
345 $biomeArray = new BiomeArray($binaryStream->get(256)); //never throws
346 }catch(BinaryDataException $e){
347 throw new CorruptedChunkException($e->getMessage(), 0, $e);
348 }
349 }
350 break;
351 case ChunkVersion::v0_9_5:
352 case ChunkVersion::v0_9_2:
353 case ChunkVersion::v0_9_0:
354 $convertedLegacyExtraData = $this->deserializeLegacyExtraData($index, $chunkVersion);
355
356 $legacyTerrain = $this->db->get($index . ChunkDataKey::LEGACY_TERRAIN);
357 if($legacyTerrain === false){
358 throw new CorruptedChunkException("Missing expected LEGACY_TERRAIN tag for format version $chunkVersion");
359 }
360 $binaryStream = new BinaryStream($legacyTerrain);
361 try{
362 $fullIds = $binaryStream->get(32768);
363 $fullData = $binaryStream->get(16384);
364 $binaryStream->get(32768); //legacy light info, discard it
365 }catch(BinaryDataException $e){
366 throw new CorruptedChunkException($e->getMessage(), 0, $e);
367 }
368
369 for($yy = 0; $yy < 8; ++$yy){
370 $storages = [SubChunkConverter::convertSubChunkFromLegacyColumn($fullIds, $fullData, $yy)];
371 if(isset($convertedLegacyExtraData[$yy])){
372 $storages[] = $convertedLegacyExtraData[$yy];
373 }
374 $subChunks[$yy] = new SubChunk(BlockLegacyIds::AIR << Block::INTERNAL_METADATA_BITS, $storages);
375 }
376
377 try{
378 $binaryStream->get(256); //heightmap, discard it
380 $unpackedBiomeArray = unpack("N*", $binaryStream->get(1024)); //unpack() will never fail here
381 $biomeArray = new BiomeArray(ChunkUtils::convertBiomeColors(array_values($unpackedBiomeArray))); //never throws
382 }catch(BinaryDataException $e){
383 throw new CorruptedChunkException($e->getMessage(), 0, $e);
384 }
385 break;
386 default:
387 //TODO: set chunks read-only so the version on disk doesn't get overwritten
388 throw new CorruptedChunkException("don't know how to decode chunk format version $chunkVersion");
389 }
390
391 $nbt = new LittleEndianNbtSerializer();
392
394 $entities = [];
395 if(($entityData = $this->db->get($index . ChunkDataKey::ENTITIES)) !== false && $entityData !== ""){
396 try{
397 $entities = array_map(fn(TreeRoot $root) => $root->mustGetCompoundTag(), $nbt->readMultiple($entityData));
398 }catch(NbtDataException $e){
399 throw new CorruptedChunkException($e->getMessage(), 0, $e);
400 }
401 }
402
404 $tiles = [];
405 if(($tileData = $this->db->get($index . ChunkDataKey::BLOCK_ENTITIES)) !== false && $tileData !== ""){
406 try{
407 $tiles = array_map(fn(TreeRoot $root) => $root->mustGetCompoundTag(), $nbt->readMultiple($tileData));
408 }catch(NbtDataException $e){
409 throw new CorruptedChunkException($e->getMessage(), 0, $e);
410 }
411 }
412
413 $finalisationChr = $this->db->get($index . ChunkDataKey::FINALIZATION);
414 if($finalisationChr !== false){
415 $finalisation = ord($finalisationChr);
416 $terrainPopulated = $finalisation === self::FINALISATION_DONE;
417 }else{ //older versions didn't have this tag
418 $terrainPopulated = true;
419 }
420
421 //TODO: tile ticks, biome states (?)
422
423 $chunk = new Chunk(
424 $subChunks,
425 $biomeArray ?? BiomeArray::fill(BiomeIds::OCEAN), //TODO: maybe missing biomes should be an error?
426 $terrainPopulated
427 );
428
429 if($hasBeenUpgraded){
430 $chunk->setTerrainDirty(); //trigger rewriting chunk to disk if it was converted from an older format
431 }
432
433 return new ChunkData($chunk, $entities, $tiles);
434 }

436 public function saveChunk(int $chunkX, int $chunkZ, ChunkData $chunkData) : void{
437 $idMap = LegacyBlockIdToStringIdMap::getInstance();
438 $index = LevelDB::chunkIndex($chunkX, $chunkZ);
439
440 $write = new \LevelDBWriteBatch();
441 $write->put($index . ChunkDataKey::OLD_VERSION, chr(self::CURRENT_LEVEL_CHUNK_VERSION));
442
443 $chunk = $chunkData->getChunk();
444 if($chunk->getTerrainDirtyFlag(Chunk::DIRTY_FLAG_BLOCKS)){
445 $subChunks = $chunk->getSubChunks();
446 foreach($subChunks as $y => $subChunk){
447 $key = $index . ChunkDataKey::SUBCHUNK . chr($y);
448 if($subChunk->isEmptyAuthoritative()){
449 $write->delete($key);
450 }else{
451 $subStream = new BinaryStream();
452 $subStream->putByte(self::CURRENT_LEVEL_SUBCHUNK_VERSION);
453
454 $layers = $subChunk->getBlockLayers();
455 $subStream->putByte(count($layers));
456 foreach($layers as $blocks){
457 if($blocks->getBitsPerBlock() !== 0){
458 $subStream->putByte($blocks->getBitsPerBlock() << 1);
459 $subStream->put($blocks->getWordArray());
460 }else{
461 //TODO: we use these in-memory, but they aren't supported on disk by the game yet
462 //polyfill them with a zero'd 1-bpb instead
463 $subStream->putByte(1 << 1);
464 $subStream->put(str_repeat("\x00", PalettedBlockArray::getExpectedWordArraySize(1)));
465 }
466
467 $palette = $blocks->getPalette();
468 $subStream->putLInt(count($palette));
469 $tags = [];
470 foreach($palette as $p){
471 $tags[] = new TreeRoot(CompoundTag::create()
472 ->setString("name", $idMap->legacyToString($p >> Block::INTERNAL_METADATA_BITS) ?? "minecraft:info_update")
473 ->setInt("oldid", $p >> Block::INTERNAL_METADATA_BITS) //PM only (debugging), vanilla doesn't have this
474 ->setShort("val", $p & Block::INTERNAL_METADATA_MASK));
475 }
476
477 $subStream->put((new LittleEndianNbtSerializer())->writeMultiple($tags));
478 }
479
480 $write->put($key, $subStream->getBuffer());
481 }
482 }
483 }
484
485 if($chunk->getTerrainDirtyFlag(Chunk::DIRTY_FLAG_BIOMES)){
486 $write->put($index . ChunkDataKey::HEIGHTMAP_AND_2D_BIOMES, str_repeat("\x00", 512) . $chunk->getBiomeIdArray());
487 }
488
489 //TODO: use this properly
490 $write->put($index . ChunkDataKey::FINALIZATION, chr($chunk->isPopulated() ? self::FINALISATION_DONE : self::FINALISATION_NEEDS_POPULATION));
491
492 $this->writeTags($chunkData->getTileNBT(), $index . ChunkDataKey::BLOCK_ENTITIES, $write);
493 $this->writeTags($chunkData->getEntityNBT(), $index . ChunkDataKey::ENTITIES, $write);
494
495 $write->delete($index . ChunkDataKey::HEIGHTMAP_AND_2D_BIOME_COLORS);
496 $write->delete($index . ChunkDataKey::LEGACY_TERRAIN);
497
498 $this->db->write($write);
499 }

506 $nbt = new LittleEndianNbtSerializer();

508 }else{
509 $write->delete($index);

529 public function getAllChunks(bool $skipCorrupted = false, ?\Logger $logger = null) : \Generator{
530 foreach($this->db->getIterator() as $key => $_){
531 if(strlen($key) === 9 && substr($key, -1) === ChunkDataKey::OLD_VERSION){
532 $chunkX = Binary::readLInt(substr($key, 0, 4));
533 $chunkZ = Binary::readLInt(substr($key, 4, 4));
534 try{
535 if(($chunk = $this->loadChunk($chunkX, $chunkZ)) !== null){
536 yield [$chunkX, $chunkZ] => $chunk;
537 }
538 }catch(CorruptedChunkException $e){
539 if(!$skipCorrupted){
540 throw $e;
541 }
542 if($logger !== null){
543 $logger->error("Skipped corrupted chunk $chunkX $chunkZ (" . $e->getMessage() . ")");
544 }
545 }
546 }
547 }
