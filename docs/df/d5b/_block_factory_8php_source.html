93 public function register(Block $block, bool $override = false) : void{
94 $typeId = $block->getTypeId();
95
96 if(!$override && isset($this->typeIndex[$typeId])){
97 throw new \InvalidArgumentException("Block ID $typeId is already used by another block, and override was not requested");
98 }
99
100 $this->typeIndex[$typeId] = clone $block;
101
102 //TODO: this bruteforce approach to discovering all valid states is very inefficient for larger state data sizes
103 //at some point we'll need to find a better way to do this
104 $bits = $block->getRequiredTypeDataBits() + $block->getRequiredStateDataBits();
105 if($bits > Block::INTERNAL_STATE_DATA_BITS){
106 throw new \InvalidArgumentException("Block state data cannot use more than " . Block::INTERNAL_STATE_DATA_BITS . " bits");
107 }
108 for($stateData = 0; $stateData < (1 << $bits); ++$stateData){
109 $v = clone $block;
110 try{
111 $v->decodeStateData($stateData);
112 if($v->computeStateData() !== $stateData){
113 //if the fullID comes back different, this is a broken state that we can't rely on; map it to default
114 throw new InvalidSerializedRuntimeDataException("Corrupted state");
115 }
116 }catch(InvalidSerializedRuntimeDataException $e){ //invalid property combination, leave it
117 continue;
118 }
119
120 $this->fillStaticArrays($v->getStateId(), $v);
121 }
122 }

124 private function fillStaticArrays(int $index, Block $block) : void{
125 $fullId = $block->getStateId();
126 if($index !== $fullId){
127 throw new AssumptionFailedError("Cannot fill static arrays for an invalid blockstate");
128 }else{
129 $this->fullList[$index] = $block;
130 $this->blastResistance[$index] = $block->getBreakInfo()->getBlastResistance();
131 $this->light[$index] = $block->getLightLevel();
132 $this->lightFilter[$index] = min(15, $block->getLightFilter() + LightUpdate::BASE_LIGHT_FILTER);
133 if($block->blocksDirectSkyLight()){
134 $this->blocksDirectSkyLight[$index] = true;
135 }
136 }

151 public function fromStateId(int $stateId) : Block{
152 if($stateId < 0){
153 throw new \InvalidArgumentException("Block state ID cannot be negative");
154 }
155 if(isset($this->fullList[$stateId])) { //hot
156 $block = clone $this->fullList[$stateId];
157 }else{
158 $typeId = $stateId >> Block::INTERNAL_STATE_DATA_BITS;
159 $stateData = $stateId & Block::INTERNAL_STATE_DATA_MASK;
160 $block = new UnknownBlock(new BID($typeId), new BlockTypeInfo(BreakInfo::instant()), $stateData);
161 }
162
163 return $block;
164 }
