59 public function __construct(string $path, \Logger $logger){
60 $this->path = $path;
61
62 if(!file_exists($this->path)){
63 $logger->debug("Resource packs path $path does not exist, creating directory");
64 mkdir($this->path);
65 }elseif(!is_dir($this->path)){
66 throw new \InvalidArgumentException("Resource packs path $path exists and is not a directory");
67 }
68
69 $resourcePacksYml = Path::join($this->path, "resource_packs.yml");
70 if(!file_exists($resourcePacksYml)){
71 copy(Path::join(\pocketmine\RESOURCE_PATH, "resource_packs.yml"), $resourcePacksYml);
72 }
73
74 $resourcePacksConfig = new Config($resourcePacksYml, Config::YAML, []);
75
76 $this->serverForceResources = (bool) $resourcePacksConfig->get("force_resources", false);
77
78 $logger->info("Loading resource packs...");
79
80 $resourceStack = $resourcePacksConfig->get("resource_stack", []);
81 if(!is_array($resourceStack)){
82 throw new \InvalidArgumentException("\"resource_stack\" key should contain a list of pack names");
83 }
84
85 foreach($resourceStack as $pos => $pack){
86 if(!is_string($pack) && !is_int($pack) && !is_float($pack)){
87 $logger->critical("Found invalid entry in resource pack list at offset $pos of type " . gettype($pack));
88 continue;
89 }
90 $pack = (string) $pack;
91 try{
92 $packPath = Path::join($this->path, $pack);
93 if(!file_exists($packPath)){
94 throw new ResourcePackException("File or directory not found");
95 }
96 if(is_dir($packPath)){
97 throw new ResourcePackException("Directory resource packs are unsupported");
98 }
99
100 $newPack = null;
101 //Detect the type of resource pack.
102 $info = new \SplFileInfo($packPath);
103 switch($info->getExtension()){
104 case "zip":
105 case "mcpack":
106 $newPack = new ZippedResourcePack($packPath);
107 break;
108 }
109
110 if($newPack instanceof ResourcePack){
111 $this->resourcePacks[] = $newPack;
112 $this->uuidList[strtolower($newPack->getPackId())] = $newPack;
113 }else{
114 throw new ResourcePackException("Format not recognized");
115 }
116 }catch(ResourcePackException $e){
117 $logger->critical("Could not load resource pack \"$pack\": " . $e->getMessage());
118 }
119 }
120
121 $logger->debug("Successfully loaded " . count($this->resourcePacks) . " resource packs");
122 }
